\begin{example}[Sample a model of a propositional formula]

    Acceptance rate:
    \begin{align*}
        \acceptanceprob
        = \frac{\contraction{\formula}}{2^{\catorder}}
    \end{align*}

    Angle:
    \begin{align*}
        \rotanglesymbol
        = 2 \cdot arcsin\left(\sqrt{\frac{\contraction{\formula}}{2^{\catorder}}}\right)
    \end{align*}

\end{example}
