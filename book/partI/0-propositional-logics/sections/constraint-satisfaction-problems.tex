\section{Constraint Satisfaction Problems}

Let us now explore a more general class of logical inference problems and discuss probabilistic entailment within that class.
We then provide further examples based on categorical constraints.
Following Chapter~5 in \cite{russell_artificial_2021}, we now define Constraint Satisfaction Problems.

\begin{definition}\label{def:csp}
    Let there be a hypergraph $\graph=(\nodes,\edges)$ and $\extnet$ be a tensor network of boolean constraint tensors $\hypercoreofat{\edge}{\catvariableof{\edge}}$ to each $\edgein$, that is
    \begin{align*}
        \extnet = \{ \hypercoreofat{\edge}{\catvariableof{\edge}} \wcols \edge\in\edges \} \, .
    \end{align*}
    We then call $\extnet$ a \emph{constraint network} and the tensor
    \begin{align*} % Necessary? Can also  directly use the contraction
        \extnetat{\nodevariables}
        = \contractionof{\extnet}{\nodevariables}
    \end{align*}
    its contraction.
    The \emph{constraint satisfaction problem (CSP)} to the constraint network $\extnet$ is the decision whether there is a state $\catindexof{\nodes}$ such that
    \begin{align*}
        \contractionof{\extnet}{\indexedcatvariableof{\nodes}} = 1 \, .
    \end{align*}
    We say the CSP is satisfiable, when there is such a state, and unsatisfiable if not.
\end{definition}

\begin{theorem}[Invariance under adding subcontractions, Rule $(i)$]
    Let there be a constraint network $\tnetof{\edges}$ on the hypergraph $\graph=(\nodes,\edges)$ and choose an arbitrary subset $\secedges\subset\edges$ and $\secnodes\subset\nodes$.
    We further choose a reduced subset $\edgesof{\mathrm{reduced}}\subset\edges$ with $\edges\backslash\secedges \subset\edgesof{\mathrm{reduced}}$ and for any $\edge\in\edges\backslash\edgesof{\mathrm{reduced}}$ we have $\edge\subset\secnodes$.
    Then we have
    \begin{align*}
        \contractionof{\tnetof{\edges}}{\nodevariables}
        =
        \contractionof{
            \tnetof{\edgesof{\mathrm{reduced}}}
            \cup \{\nzcontractionof{\tnetof{\secedges}}{\secnodes}\}
        }{\nodevariables} \, .
    \end{align*}
\end{theorem}

\begin{theorem}[Invariance under splitting disentangled edges, Rule $(ii)$]
    Let there be a constraint network $\tnetof{\edges}$ and let for some $\edgein$ with $\edge=\edgeof{0}\dot{\cup}\edgeof{1}$ exist tensors $\hypercoreofat{\edgeof{0}}{\catvariableof{\edgeof{0}}}$ and $\hypercoreofat{\edgeof{1}}{\catvariableof{\edgeof{1}}}$ obeying
    \begin{align*}
        \hypercoreofat{\edge}{\catvariableof{\edge}}
        =  \hypercoreofat{\edgeof{0}}{\catvariableof{\edgeof{0}}} \otimes  \hypercoreofat{\edgeof{1}}{\catvariableof{\edgeof{1}}} \, .
    \end{align*}
    Then
    \begin{align*}
        \contractionof{\tnetof{\edges}}{\nodevariables}
        = \contractionof{\tnetof{(\edges\cup\{\edgeof{0},\edgeof{1}\})\backslash\{\edge\}}}{\nodevariables} \, .
    \end{align*}

\end{theorem}

\subsubsection{Clauses}

An interesting class of constraint networks are those where each constraint is a clause.
Many logical inference algorithms are understood as manipulations of the constraint networks \cite{brachman_knowledge_2014}.
Before presenting the basic syntactical inference rules, let us derive contraction properties of clauses.

\begin{lemma}
    For any clause $\clauseofat{\hardparam}{\catvariableof{\hardlegset}}$ and any $\catenumerator\in\hardlegset$ and $\secheadindexof{\catenumerator}\in[2]$ we have
    \begin{align*}
        \contractionof{
            \clauseofat{\hardparam}{\catvariableof{\hardlegset}},\onehotmapofat{\headindexof{\catenumerator}}{\catvariableof{\catenumerator}}
        }{\catvariableof{\hardlegset}}
        = \begin{cases}
              \clauseof{\hardparam}{\catvariableof{\hardlegset\backslash\{\catenumerator\}}} \otimes \onehotmapofat{\headindexof{\catenumerator}}{\catvariableof{\catenumerator}}
              & \ifspace \headindexof{\catenumerator} = \secheadindexof{\catenumerator} \\
              \onesat{\catvariableof{\hardlegset\backslash\{\catenumerator\}}}\otimes \onehotmapofat{\headindexof{\catenumerator}}{\catvariableof{\catenumerator}}
              & \ifspace \headindexof{\catenumerator} \neq \secheadindexof{\catenumerator}
        \end{cases} \, .
    \end{align*}
\end{lemma}



\begin{example}[Modus ponens]
    When there is an edge
    \begin{align*}
        \edge
        = \innodes \cup \outnodes
    \end{align*}
    with
    \begin{align*}
        \hypercoreofat{\edge}{\catvariableof{\edge}}
        = \left(\bigwedge_{\node\in\innodes}\catvariableof{\node}\right) \Rightarrow \left(\bigwedge_{\node\in\outnodes}\catvariableof{\node}\right)
    \end{align*}
    and for each $\node\in\innodes$ an edge $\{\node\}$ with $\hypercoreofat{\{\node\}}{\catvariableof{\node}}=\tbasisat{\catvariableof{\node}}$, then:
    \begin{itemize}
        \item Choose $\secgraph=\left(\edge,\{\edge\}\cup\bigcup_{\node\in\innodes}\{\node\}\right)$, $\secnodes=\edge$
        \item Apply $i$ rule to replace $\secgraph$ with $\hypercoreofat{\edge}{\catvariableof{\edge}}=\bigtimes_{\node\in\edge}\tbasisat{\catvariableof{\node}}$
        \item Apply $ii$ rule to split into edges $\{\node\}$ for $\node\in\edge$ with $\tbasisat{\catvariableof{\node}}$
    \end{itemize}
\end{example}

We further derive a contraction property of arbitrary clauses with negated literals.

\begin{lemma}
    Let there be two clauses $\clauseofat{(\hardlegset^0,\headindex^0_{\hardlegset^0})}{\catvariableof{\hardlegset^0}}$, $\clauseofat{(\hardlegset^1,\headindex^1_{\hardlegset^1})}{\catvariableof{\hardlegset^1}}$ and let us assume there is $\catenumerator\in(\hardlegset^0\cap\hardlegset^1)$ with
    \begin{align*}
        \headindex^0_{\catenumerator} \neq \headindex^1_{\catenumerator} \, .
    \end{align*}
    Then
    \begin{align*}
        &\nzcontractionof{
            \clauseofat{(\hardlegset^0,\headindex^0_{\hardlegset^0})}{\catvariableof{\hardlegset^0}},
            \clauseofat{(\hardlegset^1,\headindex^1_{\hardlegset^1})}{\catvariableof{\hardlegset^1}}
        }{
            \catvariableof{(\hardlegset^0\cup\hardlegset^1)\backslash\{\catenumerator\}}
        } \\
        &\quad=
        \begin{cases}
            \onesat{\catvariableof{(\hardlegset^0\cup\hardlegset^1)\backslash\{\catenumerator\}}}
            & \ifspace \existquantwrtof{\seccatenumerator\in(\hardlegset^0\cup\hardlegset^1)\backslash\{\catenumerator\}}{\headindex^0_{\seccatenumerator}\neq\headindex^1_{\seccatenumerator}} \\
            \clauseofat{(\hardlegset^0\cup\hardlegset^1)\backslash\{\catenumerator\},\headindexof{(\hardlegset^0\cup\hardlegset^1)\backslash\{\catenumerator\}}}{
                \catvariableof{(\hardlegset^0\cup\hardlegset^1)\backslash\{\catenumerator\}}
            } & \text{else} \, .
        \end{cases}
    \end{align*}
\end{lemma}
\begin{proof}
    Since contraction is the sum of the contractions with $\tbasisat{\catvariableof{\catenumerator}}$ and $\tbasisat{\catvariableof{\catenumerator}}$, shrinking the right and the left clause.
    Can argue with the unique assignments to the left and right shrinked variables such that vanishing coordinate is possible.
    Together, if they agree, they build the unique assignment where the resolvent is vanishing.
\end{proof}


\begin{example}[Resolution]
    When there are two clauses with a complementary literal can apply $i$ with the complementary literal variable left out of the open variables and get the resolvent clause.
    Notice that the old clauses cannot always be dropped, since they are no subsets of $\secnodes$.
    This is different to the modus ponens inference rule.
\end{example}

\begin{definition}[Categorical Constraint and Atomization Variables]\label{def:catConAtomVar}
    Given a list $\catvariableof{0},\ldots,\catvariableof{\catdim-1}$ of boolean variables and a categorical variable $\catvariable$ with dimension $\catdim$ a categorical constraint is a tensor $\categoricalcoreat{\catvariableof{[\catdim]},\catvariable}$ with coordinates
    \begin{align*}
        \categoricalcoreat{\indexedcatvariableof{[\catdim]},\indexedcatvariable}
%		 \categoricalmap(\catindex,\catindexof{\variableset})
        = \begin{cases}
              1 & \ifspace \shortcatindices = \onehotmapof{\catindex} \quad \Big(\text{i.e. }\forall \catenumerator\in[\catdim]: \big(\catindex=\catenumerator\big) \Leftrightarrow \big(\catindexof{\catenumerator}=1\big) \Big)\\
              0 & \text{else} \, .
        \end{cases}
    \end{align*}
    We then call the variables  $\catvariableof{0},\ldots,\catvariableof{\catdim-1}$ the atomization variables to the categorical variable $\catvariable$.
\end{definition}

We notice that the categorical constraint tensor is the basis encoding of the one-hot map on $[\catdim]$.
The restriction to the image coordinate $\catenumerator$ of the one-hot map is the function
\begin{align*}
    \categoricalmapofat{\catenumerator}{\seccatenumerator}
    = \begin{cases}
          1 & \ifspace \seccatenumerator=\catenumerator \\
          0 & \text{else} \, .
    \end{cases}
\end{align*}
The basis encoding of the one-hot map is decomposed as
\begin{align}
    \categoricalcoreofat{\catenumerator}{\catvariableof{\catenumerator},\catvariable}
    = \tbasisat{\catvariableof{\catenumerator}} \otimes \onehotmapofat{\catenumerator}{\catvariable}
    + \onehotmapofat{0}{\catvariableof{\catenumerator}} \otimes \big(\onesat{\catvariable}- \onehotmapofat{\catenumerator}{\catvariable}\big) \, .
\end{align}
We thus have (see \figref{fig:categoricalDecomposition}) %by \theref{the:functionDecompositionBasisCP} NEED NEW REFERENCE?
\begin{align*}
    \bencodingofat{\categoricalmap}{\catvariableof{[\catdim]},\catvariable}
    = \contractionof{\left\{\categoricalcoreofat{\catenumerator}{\catvariableof{\catenumerator},\catvariable} \wcols \catindex\in[\catdim]\right\}}{\catvariable, \catvariableof{0}, \ldots, \catvariableof{\catdim-1}} \, .
\end{align*}
