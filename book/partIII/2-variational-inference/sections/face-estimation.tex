\section{Face estimation}

Generic situation:
Find the face in $\facelatticeof{\meansetof{\sstat}}$ which contains $\genmean$ in its relative interior.

\begin{theorem}
    Let $\gendistributionwith$ be a distribution with mean parameter $\genmeanwith$.
    Then the face $\faceto{\genmean}$ is the minimal face in the lattice $\facelatticeof{\meansetof{\sstat}}$ such that
    \begin{align*}
        \contraction{\gendistributionwith,\bencsstatwith,\kcoreofat{\facesymbol}{\headvariables}}
        = 1 \, .
    \end{align*}
\end{theorem}
\begin{proof}
    \begin{itemize}
        \item $\genmeanwith\in\facesymbol$ is equivalent to $\contraction{\gendistributionwith,\bencsstatwith,\kcoreofat{\facesymbol}{\headvariables}}=1$.
        \item If $\genmeanwith$ is not in the relative interior of $\facesymbol$, then it is contained in a smaller face.
    \end{itemize}
\end{proof}

\subsection{Cube-like faces}

Cube-faces are parametrized by $\hardlegset$ and a boolean tuple $\headindexof{\hardlegset}$.
For cube-like faces, it is enough to decide entailment and contradiction of each feature.

\begin{theorem}
    If the mean parameter $\genmeanwith$ to a distribution $\gendistributionwith$ is in the relative interior of a cube-like face, then the face $\faceto{\genmean}$
    \begin{align*}
        \hardlegset
        =\{\selindex \wcols \selindexin \ncond \contraction{\gendistributionwith,\sstatcoordinateofat{\selindex}{\shortcatvariables}} \in \{0,1\} \}
    \end{align*}
    and
    \begin{align*}
        \headindexof{\selindex}
        = \contraction{\gendistributionwith,\sstatcoordinateofat{\selindex}{\shortcatvariables}} \, .
    \end{align*}
\end{theorem}

\subsection{Face refinement}

We present \algoref{alg:faceRefinement} to search the $\faceto{\genmean}$ in the face lattice.
In each iteration of the \whileSymbol{} loop it either finds a facet of the current with $\genmeanwith\in\secfacesymbol$ or it terminates.
Since the face lattice has a finite depth $n$, it always terminates after at most $n$ iterations of the \whileSymbol{} loop.


\begin{algorithm}[hbt!]
    \caption{Face refinement}\label{alg:faceRefinement}
    \begin{algorithmic}
        \Require Distribution $\gendistributionwith$, statistic $\sstat$
        \Ensure Face $\faceto{\genmeanwith}$
        \iosepline
        \State $\facesymbol\algdefsymbol\genmeanset$
        \While{True}
            \State $\mathrm{refined}\algdefsymbol \falsesymbol$
            \ForAll{facets $\secfacesymbol$ of $\facesymbol$}
                \If{not $\mathrm{refined}$}
                    \State Compute
                    \begin{align*}
                        \lambda=\contraction{\gendistributionwith,\basemeasureofat{\secfacesymbol}{\shortcatvariables}}
                    \end{align*}
                    \If{$\lambda=1$}
                        \State $\mathrm{refined}\algdefsymbol \truesymbol$
                        \State $\facesymbol\algdefsymbol\secfacesymbol$
                    \EndIf
                \EndIf
            \EndFor
            \If{not $\mathrm{refined}$}
                \State \Return $\facesymbol$
            \EndIf
        \EndWhile
    \end{algorithmic}
\end{algorithm}
