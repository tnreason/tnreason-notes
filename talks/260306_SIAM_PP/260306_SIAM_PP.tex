    \section{Tensor networks for neuro-symbolic AI}

    \begin{frame}{Motivation: Toy accounting rule system}

        \begin{example}[Toy accounting rule system]
            Assume you want to book one of two accounts $\mathrm{A1}$ and $\mathrm{A2}$ depending on a feature $\mathrm{F}$ of an invoice, based on two rules:
            \begin{itemize}
                \item \textcolor{\concolor}{Exactly one account must be booked.}
                \item \textcolor{\probcolor}{If feature $\mathrm{F}$ is present on the invoice, the account $\mathrm{A1}$ is typically booked.}
            \end{itemize}
        \end{example}

        \begin{itemize}
            \item {Accounting rules} are formalised in propositional logics, triggered by receipts properties.
            \item These rules can either be \textcolor{\concolor}{hard restrictions}, or \textcolor{\probcolor}{prone to exceptions}.
        \end{itemize}

        \begin{center}
            \bf We apply tensor networks for neuro-symbolic use cases like this!
        \end{center}

    \end{frame}

    \begin{frame}{Tensors: Natural description of factored representations}

    {\bf Factored representations}
        are collection of variables $\shortcatvariables$, each with a finite set of states $[\catdimof{\catenumerator}]=\{0,\ldots,\catdimof{\catenumerator}-1\}$
        \cite{russell_artificial_2021}.
        The set of states is then the cartesian product
        \begin{align*}
            \facstates
        \end{align*}

            {\bf Knowledge about the system}, for example the probability of each state, is a \emph{tensor}
        \begin{align*}
            \hypercorewith \defcols \facstates \rightarrow \rr
        \end{align*}
        We represent them by diagrams with legs decorated by the variables:
        \begin{center}

            \begin{tikzpicture}[scale=0.35,thick] % , baseline = -3.5pt

    \end{frame}

    \begin{frame}{Sparsity by tensor network decompositions}

        We represent functions between factored state sets by their \emph{basis encodings}
        \begin{align*}
            \bencodingofat{\exformula}{\headvariables,\shortcatvariables}
            = \sum_{\shortcatindicesin} \onehotmapofat{\exformula(\shortcatindices)}{\headvariables}
            \otimes \onehotmapofat{\shortcatindices}{\shortcatvariables}
        \end{align*}

        \begin{block}{Main sparsity principle}
            Any decomposition of a function is reflected in a tensor network decomposition of the basis encoding.
        \end{block}

    \section{Quantum inference for neuro-symbolic AI}

    \begin{frame}{Motivation for quantum inference}
        \begin{block}{Efficiency problem}
            Tensor network contractions are the workhorse of inference, but they are often computationally demanding.
        \end{block}

        Mitigate computational cost of contractions:
        \begin{center}
            \bf Get square root speedups through quantum inference.
        \end{center}

        The similarity of quantum circuits and tensor networks allows to apply quantum algorithms for inference tasks.

        \begin{block}{Quantum Rejection Sampling \cite{low_quantum_2014}}
            Square root advantage over classical rejection sampling schemes.
        \end{block}

    \end{frame}

    \begin{frame}{From tensor networks to quantum circuits}

        The \emph{oracle} to a boolean functions $\exformula$ is the controlled unitary with slices to $\shortcatindices\in[2]^{\catorder}$
        \begin{align*}
            \qcbencodingofat{\exformula}{\cheadvariableof{\exformula}{\insymbol}=0,\cheadvariableof{\exformula}{\outsymbol},\shortcatvariables=\shortcatindices}
            = \onehotmapofat{\exformula(\shortcatindices)}{\cheadvariableof{\exformula}{\outsymbol}} \, .
        \end{align*}
            {\bf Connection:} Oracles relate with basis encodings of functions as:
        \begin{center}
            \input{tikz-pics/bencoding-computation-circuit}
        \end{center}
        The same sparsity principles can thus be applied to get efficient quantum circuits for the oracles.%, and thus for the Grover operator.
    \end{frame}

    {\bf Approach:}

    \end{frame}

    \begin{frame}{Summary and Outlook}

    {\bf Tensor networks}
        \begin{itemize}
            \item Tensors appear naturally as factored system representations
            \item \emph{Sparsity principle:} Decomposition into tensor networks
            \item \emph{Inference principle:} Tensor network contractions
        \end{itemize}

            {\bf Quantum circuits}
        \begin{itemize}
            \item Follow the same formalism and decomposition principles
            \item Allow for square root speedups for inference tasks
        \end{itemize}

            {\bf Implementations} in \python{} packages:
        \begin{itemize}
            \item {\bf \tnreason{}}\footnote{https://pypi.org/project/tnreason/}: Tensor network based reasoning
            \item {\bf \qcreason{}}\footnote{https://pypi.org/project/qcreason/}: Quantum inference for contractions
        \end{itemize}

    \end{frame}
