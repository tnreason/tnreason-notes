    \section{Tensor networks in neuro-symbolic AI}

    \begin{frame}{Motivation: Accounting Rule Systems}

        \begin{example}[Toy Accounting Rule System]
            Assume you want to book one of two accounts $\mathrm{A1}$ and $\mathrm{A2}$ depending on a feature $\mathrm{F}$ of an invoice, based on two rules:
            \begin{itemize}
                \item \textcolor{\concolor}{Exactly one account must be booked.}
                \item \textcolor{\probcolor}{If feature $\mathrm{F}$ is present on the invoice, the account $\mathrm{A1}$ is typically booked.}
            \end{itemize}
        \end{example}

        \begin{itemize}
            \item {Accounting rules} are formalised in \textbf{propositional logics}, triggered by receipts properties.
            \item These rules can either be \textcolor{\concolor}{hard restrictions}, or \textcolor{\probcolor}{prone to exceptions}.
        \end{itemize}
    \end{frame}


    \begin{frame}{Tensors: A natural description of factored systems}

    {\bf Factored System [Russel and Norvig, 2021]:}
        Collection of variables, each with a finite set of states.

            {\bf Knowledge about the system:}
        Store an information (e.g. Boolean) to each system state, and get \emph{Tensors}


    \end{frame}

    \begin{frame}{Model Sparsity by Tensor Network Decompositions}

        Basis encodings of functions between factored systems are tensors
        \begin{align*}
            \bencodingofat{\exfunction}{\headvariables,\shortcatvariables}
            = \sum_{\shortcatindicesin} \onehotmapofat{\exfunction(\shortcatindices)}{\headvariables}
            \otimes \onehotmapofat{\shortcatindices}{\shortcatvariables}
        \end{align*}

        \begin{block}{Main Sparsity Principle}
            Any decomposition of the function is reflected in a tensor network decomposition.
        \end{block}

    \begin{frame}{Inference by Contractions}

        \begin{block}{Main Reasoning Principle}
            Model counts and marginal probabilities are contractions of tensor networks representing a model.
        \end{block}

        More advanced inference schemes:
        \begin{itemize}
            \item Parameter estimation: Moments in an inner loops, which are iteratively matched.
            \item Expectation propagation: Approximation framework for contractions by orchestration of efficient inference steps.
        \end{itemize}

    \end{frame}
