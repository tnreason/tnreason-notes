        \question Let us consider the propositional formula
        \begin{align*}
            \formulaat{\catvariableof{0},\catvariableof{1}}
            =
            \coloredmatrixof{
                0 & 1 \\
                1 & 0
            }{\catvariableof{0},\catvariableof{1}} \, .
        \end{align*}
        How many coordinates does this tensor have? \\
        \begin{oneparcheckboxes}
            \choice 4
            \choice 16
            \choice 2
        \end{oneparcheckboxes}\\
        What is the order of this tensor? \\
        \begin{oneparcheckboxes}
            \choice 4
            \choice 16
            \choice 2
        \end{oneparcheckboxes}

            \choice $2^{(2^{\catorder})}$

        \begin{align*}

            }{\catvariableof{0},\catvariableof{1}}

            \formulaofat{1}{\catvariableof{0},\catvariableof{1}}
