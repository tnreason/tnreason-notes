    \section{Log-likelihood}


    \begin{frame}{How to select a model given data?}
        Let us consider the typical \emph{learning problem}:
        \begin{itemize}
            \item The distribution $\probwith$ of variables $\shortcatvariables$ is unknown
            \item We observe $\datdim$ samples
            \begin{align*}
                \shortcatindices^{\datindex} = (\catindexof{0}^{\datindex},\ldots,\catindexof{\catorder-1}^{\datindex})
            \end{align*}
            enumerated by $\datindexin$.
        \end{itemize}

        We assume that each sample $\shortcatindices^{\datindex}$ is drawn
        \begin{itemize}
            \item \emph{i}ndependently from each other and
            \item \emph{i}dentically \emph{d}istributed by a distribution $\probwith$.
        \end{itemize}
        This assumption is abbreviated by \emph{i.i.d.}.
        \\
        The probability of drawing a dataset $\shortcatvariables^{[\datdim]}$ from a distribution $\probwith$ is then
        \begin{align*}
            \prod_{\datindexin} \probat{\shortcatvariables=\shortcatindices^{\datindex}} \, .
        \end{align*}
        This is the \emph{likelihood} of $\probwith$.
    \end{frame}


    \begin{frame}{The likelihood principle}

        We make the hypothesis that the distribution is from a set $\Gamma$ of possible distributions.
        A natural choice would be to choose the distribution maximizing the likelihood
        \begin{align*}
            \argmax_{\probwith\in\Gamma}\left(\prod_{\datindexin} \probat{\shortcatvariables=\shortcatindices^{\datindex}}\right) \, .
        \end{align*}
        Sometimes we \emph{regularize}, that is prefer simple distributions over more complex ones.

        \begin{block}{Likelihood principle}
            The selection of a distribution should depend on the data $\shortcatvariables^{[\datdim]}$ only through the likelihood
            \begin{align*}
                \prod_{\datindexin} \probat{\shortcatvariables=\shortcatindices^{\datindex}} \, .
            \end{align*}
        \end{block}
    \end{frame}

    \begin{frame}{Negative log-likelihood}
        We simplifiy the likelihood by taking the log and averaging.
        This is the \emph{log-likelihood}
        \begin{align*}
            \frac{1}{\datdim} \lnof{\prod_{\datindexin} \probat{\shortcatindices^{\datindex}}}
            = \frac{1}{\datdim} \sum_{\datindexin} \lnof{\probat{\shortcatindices^{\datindex}}} \, .
        \end{align*}

        The maximum likelihood method corresponds then with the minimization of the \emph{negative log-likelihood (NLL)}
        \begin{align*}
            \argmin_{\probwith\in\Gamma}\sum_{\datindexin}
            -\lnof{\probat{\shortcatvariables=\shortcatindices^{\datindex}}} \, .
        \end{align*}

    \end{frame}


    \section{Empirical Distributions}

    \begin{frame}{Empirical distributions}

        Let us understand the uniform choice of a sample from $\shortcatindices^{[\datdim]}$ as a distribution
        \begin{align*}
            \empdistributionat{\shortcatvariables}
            = \frac{1}{\datdim} \sum_{\datindexin} \onehotmapofat{\shortcatindices^{\datindex}}{\shortcatvariables}
        \end{align*}
        which we call \emph{empirical distribution}.

        We have the $\cpformat$ decomposition of the empirical distribution
        \input{tikz/emp-distribution}
        where
        \begin{align*}
            \probofat{\catenumerator}{\catvariableof{\catenumerator}|\datvariable}
            = \sum_{\datindexin}
            \onehotmapofat{\catindexof{\catenumerator}^{\datindex}}{\catvariableof{\catenumerator}} \otimes
            \onehotmapofat{\datindex}{\datvariable} \, .
        \end{align*}

    \end{frame}

    \begin{frame}{Perspective of basis calculus}

        We understand the empirical distribution to samples $\shortcatindices^{[\datdim]}$ as the basis encoding of the sample selecting map
        \begin{align*}
            \datamap \defcols [\datdim] \rightarrow \bigtimes_{\catenumeratorin} [\catdimof{\catenumerator}]
        \end{align*}
        defined for $\datindexin$ by $\datamap(\datindex) = \shortcatindices^{\datindex}$.
        To be more precise we have
        \begin{align*}
            \empdistributionat{\shortcatvariables} =
            \contractionof{\bencodingofat{\datamap}{\shortcatvariables,\datvariable},\frac{1}{\datdim}\onesat{\datvariable}}{\shortcatvariables} \, .
        \end{align*}

        The $\cpformat$ decomposition of the empirical distribution is then based on the decomposition of the basis encoding into the basis encodings of its coordinate restrictions:

        \input{tikz/emp-distribution-bc}

    \end{frame}


    \begin{frame}{Cross-entropy}

        The \emph{cross-entropy} between two distributions $\secprobat{\shortcatvariables}$ and $\probwith$ is
        \begin{align*}
            \centropyof{\secprobtensor}{\probtensor}
            = \contraction{\secprobat{\shortcatvariables},-\lnof{\probwith}} \, .
        \end{align*}

        The negative log-likelihood is the cross-entropy between the empirical distribution and the candidate $\probwith$.

    \end{frame}


    \section{Learning Bayesian networks}

    \begin{frame}{Two learning tasks}
        \begin{itemize}
            \item Structure learning: Find the hypergraph $\graph$
            \item Parameter estimation: Find the best decorations by conditional probability distributions
        \end{itemize}

    \end{frame}

    \begin{frame}{Decomposing the negative log-likelihood}
        For a Bayesian network we have for any index tuple $\shortcatindices$
        \begin{align*}
            \lnof{\probat{\shortcatvariables=\shortcatindices}}
            = \sum_{\catenumeratorin}\lnof{\condprobat{\indexedcatvariableof{\catenumerator}}{\indexedcatvariableof{\parentsof{\catenumerator}}}} \, .
        \end{align*}
        The NLL thus decomposes into local terms
%        \begin{align*}
%            &\frac{1}{\datdim}\sum_{\datindexin} -\lnof{\probat{\shortcatvariables=\shortcatindices^{\datindex}}} \\
%            &\quad= \frac{1}{\datdim}\sum_{\datindexin}\sum_{\catenumeratorin} -\lnof{\condprobat{\indexedcatvariableof{\catenumerator}}{\indexedcatvariableof{\parentsof{\catenumerator}}}} \\
%            &\quad= \sum_{\catenumeratorin}\left(\frac{1}{\datdim}\sum_{\datindexin}-\lnof{\condprobat{\indexedcatvariableof{\catenumerator}^{\datindex}}{\indexedcatvariableof{\parentsof{\catenumerator}}^{\datindex}}}\right)
%        \end{align*}
        \begin{align*}
            &\contraction{\empdistributionwith,-\lnof{\probat{\shortcatvariables}}} \\
            &\quad= \contraction{\empdistributionwith,\sum_{\catenumeratorin} -\lnof{\condprobat{\catvariableof{\catenumerator}}{\catvariableof{\parentsof{\catenumerator}}}}} \\
            &\quad= \sum_{\catenumeratorin}\left(\contraction{\empdistributionwith,-\lnof{\condprobat{\indexedcatvariableof{\catenumerator}}{\indexedcatvariableof{\parentsof{\catenumerator}}}}}\right)
        \end{align*}

        The minimum can found separately for each $\catenumerator$ by the minimization problem
        \begin{align*}
            \argmin_{\condprobat{\catvariableof{\catenumerator}}{\catvariableof{\parentsof{\catenumerator}}}}
            \frac{1}{\datdim}\sum_{\datindexin}-\lnof{\condprobat{\indexedcatvariableof{\catenumerator}}{\indexedcatvariableof{\parentsof{\catenumerator}}}} \, .
        \end{align*}
    \end{frame}

    \begin{frame}{Optimum by normalization of the empirical distribution}
        The optima for each conditional distribution is taken at
        \begin{align*}
            &\argmin_{\condprobat{\catvariableof{\catenumerator}}{\catvariableof{\parentsof{\catenumerator}}}}
            \contraction{\empdistribution,-\lnof{\condprobat{\indexedcatvariableof{\catenumerator}}{\indexedcatvariableof{\parentsof{\catenumerator}}}}} \\
            &\quad= \normalizationofwrt{\empdistributionwith}{\catvariableof{\catenumerator}}{\catvariableof{\parentsof{\catenumerator}}}
        \end{align*}
        This can be verified by the first order criterion (vanishing gradients).
    \end{frame}


    \begin{frame}{Parameter estimation}


        \begin{block}{Parameter estimation for Bayesian networks}
            \begin{itemize}
                \item The likelihood of Bayesian networks decomposes into local terms.
                \item Best log-likelihood maximizers are the conditional probability distributions.
            \end{itemize}
        \end{block}

    \end{frame}

\begin{frame}{Observations of a Factored System}

    \begin{example}[Students and their semester]
	We ask students about their semester and want to capture the statistics.
    \end{example}

    \begin{block}{Task: Representation of Observations}
        Let there be observations of the system by a set of states
        \[ \{(\catindicesof{\datindex}) \, : \, \datindexin \} \, . \]
        How to capture them by tensors?
    \end{block}

    Aim:
    \begin{itemize}
        \item Represent the observations for querying
        \item Find models making sense out of the observations
    \end{itemize}

\end{frame}


\begin{frame}{Formalization with a sample selector map}

    Given a dataset $\{(\catindicesof{\datindex}) \, : \, \datindexin \}$ of samples of the factored system we define the sample selector map
    \[ \datacore : [\datanum] \rightarrow \facstates \]

    \medskip

    \begin{block}{Idea}
        Enrich the factored system with variables $\catvariables$ by a data selection variable $\aselectionvariable$ and encode $\datacore$ using the one-hot encoding of the larger system.
    \end{block}


\end{frame}


\begin{frame}{More general: Encoding of maps between Factored Systems}

    \begin{definition}
        Let $\exfunction$ be a function
        \[ \exfunction : \facstates \rightarrow  \secfacstates \]
        which maps the states of a factored system to variables $\catvariables$ to the states of another factored system with variables $\seccatvariables$.
        Then the tensor representation of $\exfunction$ is a tensor
        \[ \bencodingof{\exformula} \in \left(\exfunctiontargetspace\right)  \otimes \left(\facspace\right) \]
        defined by
        \[ \bencodingof{\exformula} = \sum_{\catindexofin{0}} \cdots  \sum_{\catindexofin{\atomenumerator-1}}  \onehotmapof{\exfunction(\catindices)} \otimes \onehotmapof{\catindices} \, . \]
    \end{definition}

\end{frame}

\begin{frame}{More efficient data representation}

    The storage demand of $\datacore$ \emph{increases exponentially }
    \[ \mathrm{dim}\left[\datacore\right] = \left( \facdim \right) \cdot \datanum \, . \]

    \begin{block}{Store the values $\catvariableof{\atomenumerator}$ separately for each $\atomenumeratorin$!}

        \medskip

        The collection of encoding tensors
        \[ \datacoreof{\atomenumerator} \in \rr^{\catdimof{\atomenumerator}\times\datanum} \]
        to the maps
        \[ \datamap_{\atomenumerator} :[\datanum] \rightarrow [\catdimof{\atomenumerator}] \quad , \quad  \datamap_{\atomenumerator}(\datindex) =  \catindex^{\datindex}_{\atomenumerator}\]
%defined by
%	\[ \datacoreof{\atomenumerator}_{: \datindex} =\onehotmapof{\catindexof{\atomenumerator}^\datindex} \]
        has a \emph{linearly increasing} storage demand of
        \[ \sum_{\atomenumeratorin} \mathrm{dim}\left[\datacoreof{\atomenumerator}\right] = \sum_{\atomenumeratorin} \catdimof{\atomenumerator}\cdot \datanum \, .  \]

    \end{block}
\end{frame}
