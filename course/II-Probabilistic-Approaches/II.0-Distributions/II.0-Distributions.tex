        \begin{align*}
            \probat{\catvariableof{A1}=1} = \frac{2}{3} \quad > \quad   \probat{\catvariableof{A2}=1} = \frac{1}{3} \, .
        \end{align*}

    \begin{frame}{Uncertainty: Summarizing the unknown}

        Uncertainties need to be handled:
        \begin{itemize}
            \item Not all variables are measured. \\
            Example: When we do not know the value of $\var{Feature}$, then we do not know whether $\var{Account 1}$ or $\var{Account 2}$ have to be booked.
            \item Parts of the models are not known. \\
            Example: Even if $\var{Feature}$ is known to be $\falsesymbol$, then we do not know whether $\var{Account 1}$ or $\var{Account 2}$ have to be booked.
        \end{itemize}

    \begin{frame}{Probability distributions}

        Instead of storing 0/1 for possibilities, we can store probabilities (in the interval $[0,1]$) in the coordinates of a tensor.

        The total probability of all states must be $1$ and can be calculated by contracting all legs of the tensor since
        \begin{align*}
            \sum_{\shortcatindices} \probat{\indexedshortcatvariables}
            = \contraction{\probwith} \, .
        \end{align*}

        \begin{definition}[Probability Tensor]
        	Let there be a factored system $\facsystem$ defined by variables $\catvariableof{\atomenumerator}$ taking values in $[\catdimof{\atomenumerator}]$.
            A probability distribution over the states of $\facsystem$ is a tensor $\probwith$ with coordinates in the interval $[0,1]$ such that
            \begin{align*}
                \contraction{\probwith} = 1 \, .
            \end{align*}
        \end{definition}

    \end{frame}


    \begin{frame}{Directed hyperedges}

        Let us introduce \emph{directed hyperedges}:
        Instead of just being subsets of the variables, directed hyperedges are pairs of incoming variables $\incomingnodes\subset\nodes$ and outgoing variables $\outgoingnodes\subset\nodes$.
        \medskip \\
        To highlight the normalization constraint we understand probability distributions as tensors on directed hyperedges, where all variables are outgoing:

        Incoming variables will be used later to highlight conditioning.

    \end{frame}

    \begin{frame}{Marginal distributions}

        Summation of probabilities of states are done by contractions of the probability distribution tensor.

        \begin{definition}[Marginal distribution]
        Given a probability distribution $\probwith$ and a subset $\arbset\subset[\catorder]$, the marginal distribution of the variables $\catvariableof{\arbset}$ is the distribution
        \begin{align*}
            \probat{\catvariableof{\arbset}}
            = \contractionof{\probwith}{\catvariableof{\arbset}} \, .
        \end{align*}
        \end{definition}

    \begin{frame}{Normalizations}

        More general, conditionalization amounts to normalization of probability tensors.

        \begin{definition}[Normalization]
            The \emph{normalization} of a tensor $\hypercorewith$ with respect to disjoint subsets $\outnodes,\innodes\subset[\catorder]$ is the tensor
            \begin{align*}
                \normalizationofwrt{\hypercorewith}{\catvariableof{\outnodes}}{\catvariableof{\innodes}}
            \end{align*}
            defined by the coordinates
            \begin{align*}
                &\normalizationofwrt{\hypercorewith}{\indexedcatvariableof{\outnodes}}{\indexedcatvariableof{\innodes}} \\
                &\quad = \begin{cases}
                             \frac{
                                 \contractionof{\hypercorewith}{\indexedcatvariableof{\outnodes},\indexedcatvariableof{\innodes}}
                             }{
                                 \contractionof{\hypercorewith}{\indexedcatvariableof{\innodes}}
                             } & \ifspace \contractionof{\hypercorewith}{\indexedcatvariableof{\innodes}}\neq 0 \\
                             \frac{1}{\prod_{\node\in\outnodes}\catdimof{\node}} &  \text{else}
                \end{cases}
            \end{align*}
        \end{definition}

    \end{frame}


    \begin{frame}{Normalized tensors and directed notation}

        Directed tensors represent normalizations: Each slice of the incoming variables is a normalized tensor, i.e. coordinates summing to $1$.

        For example a probability distribution is normalized without incoming variables since
        \begin{align*}
            \normalizationof{\probwith}{\shortcatvariables}
            = \probwith
        \end{align*}
        This is denoted by
        \stantikzpic{
            \draw (-3,-1) rectangle (3,1);
            \drawtextnode{0}{0}{\corelabelsize $\probwith$}
            \draw[->-] (-2,-1) -- (-2,-2.5) node[midway,left] {\colorlabelsize $\catvariableof{0}$};
            \drawtextnode{0}{-1.75}{$\cdots$}
            \draw[->-] (2,-1) -- (2,-2.5) node[midway,right] {\colorlabelsize $\catvariableof{\catorder-1}$};
        }
        A conditional distribution is invariant under further normalizations (with the same incoming variables) and is thus denoted by
        \stantikzpic{

            \drawtextnode{0}{-1.5}{\colorlabelsize $\cdots$}

    \end{frame}


    \begin{frame}{Formal Definition of Conditional Distributions}

        \begin{definition}[Conditional Probability]
	Given a distribution $\probwith$ and disjoint subsets $\outnodes,\innodes\subset[\catorder]$, the distribution of $\catvariableof{\outnodes}$ conditioned on $\catvariableof{\innodes}$ is the normalization
    \begin{align*}
        \condprobof{\catvariableof{\outnodes}}{\catvariableof{\innodes}}
        = \normalizationofwrt{\probwith}{\catvariableof{\outnodes}}{\catvariableof{\innodes}}
    \end{align*}
        \end{definition}

    \end{frame}

    \begin{frame}{Events and their probability}
        \par
        \emph{Events} are subsets
        \begin{align*}
            \arbset \subset \bigtimes_{\catenumeratorin} [\catdimof{\catenumerator}]
        \end{align*}
        of the possible samples.
        When the leg dimensions are $2$ we build the formula
        \begin{align*}
            \formulawith
            = \sum_{\shortcatindices\in\arbset} \onehotmapofat{\shortcatindices}{\shortcatvariables} \, .
        \end{align*}
        Probability of the event is then the contraction
        \begin{align*}
            \sum_{\shortcatindices\wcols\formulaat{\indexedshortcatvariables}=1} \probat{\indexedshortcatvariables}
            = \contraction{\probwith,\formulawith} \, .
        \end{align*}
    \end{frame}

    \begin{frame}{Comparing logics and probability theory}

        \begin{block}{Modelling assumptions}
            \begin{itemize}
                \item \textbf{Ontological}: What properties does a system have? \\
                \hspace{0.5cm} $\rightarrow$ We assume \emph{fixed sets of categorical properties}.
                \item \textbf{Epistemological}: What can we tell about these properties? \\
                \hspace{0.5cm} $\rightarrow$ We associate \emph{possibilities or probabilities} with states.
            \end{itemize}
        \end{block}

        Two main frameworks differing in the \emph{empistemological} assumptions:
        \begin{itemize}
            \item Logics (\emph{Possibilities}): The traditional mathematical model of intelligence.
            \item Probability Theory (\emph{Probabilities}): Handling uncertainties about the systems state.
        \end{itemize}
        In both cases, we encode our knowledge in an arrangement of numbers:
        \emph{Tensors} with efficient representations by \emph{Tensor Networks}.

    \end{frame}
