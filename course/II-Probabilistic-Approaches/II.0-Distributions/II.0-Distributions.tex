    \section{Uncertainties}

    \begin{frame}{Brittleness of Logical Reasoning}

        Logics cannot handle uncertainties.

        But: They are already built in model counts.


    \end{frame}


    \begin{frame}{Uncertainty: Summarizing the unknown}

        Model of uncertainties: Not all variables are measureable.

        Example: Weather.

    \end{frame}

    \begin{frame}{Example}

        Three variables: \emph{Cloud seen}, \emph{Rain today} and \emph{Rain tomorrow}.
        Let us think of a deterministic relation:
        \begin{itemize}
            \item If we see a cloud, it will rain today and tomorrow.
            \item If we do not see a cloud, it will not rain today and tomorrow.
        \end{itemize}

        Now, cloud is not observable, but we want to reason about the probability of rain today and tomorrow.

        Logics: Fails, since no entailment of rain today and tomorrow.
        Probability theory: Works, since we can assign probabilities to the states of rain today and tomorrow, resulting from averages of cloud and not cloud.

    \end{frame}


    \begin{frame}{Comparing logics and probability theory}

        \begin{block}{Modelling assumptions}
            \begin{itemize}
                \item \textbf{Ontological}: What properties does a system have? \\
                \hspace{0.5cm} $\rightarrow$ We assume \emph{fixed sets of categorical properties}.
                \item \textbf{Epistemological}: What can we tell about these properties? \\
                \hspace{0.5cm} $\rightarrow$ We associate \emph{possibilities or probabilities} with states.
            \end{itemize}
        \end{block}

        Two main frameworks differing in the \emph{empistemological} assumptions:
        \begin{itemize}
            \item Logics (\emph{Possibilities}): The traditional mathematical model of intelligence.
            \item Probability Theory (\emph{Probabilities}): Handling uncertainties about the systems state.
        \end{itemize}
        In both cases, we encode our knowledge in an arrangement of numbers:
        \emph{Tensors} with efficient representations by \emph{Tensor Networks}.

    \end{frame}

    \section{Probability Distributions}


    \begin{frame}{Recap: Tensors as Propositional Formulas}

%        \begin{tabular}[c|c]
%            Tensor & Propositional Formula
%        \end{tabular}

        So far: Tensors with Boolean coordinates and leg dimension $2$

        Generalize to
        \begin{itemize}
            \item arbitrary leg dimensions
            \item real valued coordinates instead of Booleans $\{0,1\}$
        \end{itemize}
    \end{frame}


    \begin{frame}{Probability distributions}

        Instead of storing 0/1 for possibilities, we can store probabilities (in the interval $[0,1]$) in the coordinates of a tensor.

    \end{frame}



    \begin{frame}{Normalization constraint}

        Probability distributions sum to $1$.

        We depict this by outgoing legs.

    \end{frame}
