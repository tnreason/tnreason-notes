    \begin{frame}{Reasoning about the future and the past}

        {\bf Intuitive:} Conditioning on the past we reason on the future:
        \begin{align*}
            \contractionof{\onehotmapofat{\catindexof{\catenumerator}}{\catvariableof{\catenumerator}},
                \condprobat{\catvariableof{\catenumerator+1}}{\catvariableof{\catenumerator}}
            }{\catvariableof{\catenumerator+1}}
            =
            &0.5 \cdot \onehotmapofat{\catindexof{\catenumerator}}{\catvariableof{\catenumerator+1}} \\
            &+ 0.25 \cdot \onehotmapofat{\catindexof{\catenumerator}-1}{\catvariableof{\catenumerator+1}}
            + 0.25 \cdot \onehotmapofat{\catindexof{\catenumerator}+1}{\catvariableof{\catenumerator+1}}
        \end{align*}

            {\bf Counter-intuitive:} Conditioning on the future we can also reason on the past:

        \begin{align*}
            \contractionof{
                \condprobat{\catvariableof{\catenumerator+1}}{\catvariableof{\catenumerator}},
                \onehotmapofat{\catindexof{\catenumerator}}{\catvariableof{\catenumerator+1}}
            }{\catvariableof{\catenumerator}}
            =
            &0.5 \cdot \onehotmapofat{\catindexof{\catenumerator}}{\catvariableof{\catenumerator}} \\
            &+ 0.25 \cdot \onehotmapofat{\catindexof{\catenumerator}-1}{\catvariableof{\catenumerator}}
            + 0.25 \cdot \onehotmapofat{\catindexof{\catenumerator}+1}{\catvariableof{\catenumerator}}
        \end{align*}
        \begin{center}
            \bf How to distinguish between these directions?
        \end{center}

    \end{frame}

    \begin{frame}{Interventions}

        \begin{block}{Interventional reasoning}
            AI is not only about modelling the observed, but reasoning about consequences:
            When we intervene on a variable, what would be the effect on the further variables?
        \end{block}

        This is necessary in many realistic situations:
        \begin{itemize}
            \item Will product sales increase if we lower the price?
            \item Will I pass the exam tomorrow if I study today?
        \end{itemize}

            {\bf Idea:}
        \begin{itemize}
            \item Understand the directed hyperedges to model causal consequences.
            \item But: How to deal with the ambiguity?
        \end{itemize}

        \begin{center}
            \bf Associations in Bayesian Networks can not model interventions!
        \end{center}
    \end{frame}


    \begin{frame}{Do variables: Modelling interventions}

        We add to each variable $\catvariableof{\catenumerator}$ a \emph{do variable $\dovariableof{\catenumerator}$} of dimension $[\catdimof{\catenumerator}+1]$ modelling the invervention on it:
        \begin{itemize}
            \item Intervention on the variable $\catvariableof{\catenumerator}$ when $\doindexof{\catenumerator}\in[\catdimof{\catenumerator}]$
            \item No intervention on the variable when $\doindexof{\catenumerator}=\catdimof{\catenumerator}$
        \end{itemize}

        This is realized by a tensor
        \begin{align*}
            \condprobwrtof{\causalsymbol}{\catvariableof{\catenumerator}}{\catvariableof{\parentsof{\catenumerator}},\dovariableof{\catenumerator}}
            &= \sum_{\catindexofin{\catenumerator}} \onehotmapofat{\catindexof{\catenumerator}}{\catvariableof{\catenumerator}}
            \otimes \onesat{\catvariableof{\parentsof{\catenumerator}}}
            \otimes  \onehotmapofat{\catindexof{\catenumerator}}{\dovariableof{\catenumerator}} \quad \text{(intervention)} \\
            & \quad + \condprobat{\catvariableof{\catenumerator}}{\catvariableof{\parentsof{\catenumerator}}}
            \otimes \onehotmapofat{\catdimof{\catenumerator}}{\dovariableof{\catenumerator}} \quad \text{(no intervention)}
        \end{align*}

        This resolves the ambiguity in Bayesian network representation!

    \end{frame}

    \begin{frame}{Counterfactuals}

        \begin{block}{Counterfactual reasoning}
            \begin{itemize}
                \item A particular state has been observed.
                \item We reason about a hypothetical world.
            \end{itemize}
        \end{block}

        This appears in important situation:
        \begin{itemize}
            \item Would the accident have happened had the safety protocol been respected?
            \item Would the person have developed cancer had he stopped smoking earlier?
        \end{itemize}

        \begin{center}
            \bf Interventions are not enough for these questions!
        \end{center}
    \end{frame}


    \begin{frame}{Response variables}

        \begin{block}{Interpretation of randomness}
            Random behavior in observed variables is due to exogenous variables, which states are not observed.
        \end{block}

        To decompose each conditional distribution we introduce:
        \begin{itemize}
            \item Response variables $\selvariableof{\catenumerator}$ summarize the exogenous variables
            \item Function $\exfunctionof{\catenumerator}$ describing a deterministic dependence of $\catindexof{\catenumerator}$ on the parent states $\catindexof{\parentsof{\catenumerator}}$ and the state $\selindexof{\catenumerator}$ of the response variable
        \end{itemize}
        The conditional probabilities are then contracted basis encodings, that is
        \stantikzpic{
            \drawtdwire{0}{3}{$\catvariableof{\catenumerator}$}
            \drawtensorblock{0}{0}{3}{$\condprobat{\catvariableof{\catenumerator}}{\catvariableof{\parentsof{\catenumerator}}}$}
            \drawtdwire{-2}{-1}{}
            \drawtextnode{0}{-2}{\colorlabelsize $\catvariableof{\parentsof{\catenumerator}}$ }
            \drawtdwire{2}{-1}{}

            \drawtextnode{5}{0}{${=}$}
            \begin{scope}[shift={(10,0)}]
                \drawtdwire{0}{3}{$\catvariableof{\catenumerator}$}
                \drawtensorblock{0}{0}{3}{$\bencodingof{\exfunctionof{\atomenumerator}}$}
                \drawtdwire{-2}{-1}{}
                \drawtextnode{0}{-2}{ \colorlabelsize $\catvariableof{\parentsof{\catenumerator}}$ }
                \drawtdwire{2}{-1}{}

                \drawldwire{3}{0}{$\selvariableof{\catenumerator}$}
                \drawtensorblock{7}{0}{2}{$\probat{\selvariableof{\catenumerator}}$}
            \end{scope}
            \drawtextnode{20}{-2}{${.}$}
        }
    \end{frame}
