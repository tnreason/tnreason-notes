    \section{Independence of variables}


    \begin{frame}{How to beat the curse of dimensions?}

        Probability distribution of factored systems with states $\facstates$ has
        \[\left(\prod_{\atomenumeratorin}\catdimof{\atomenumerator}\right) -1 \]
        degrees of freedom (coordinates to specify and store).

        \medskip

        Mitigation: \emph{Tensor Network Decompositions}

        \begin{block}{Independencies of Random Variables}
            Decompositions of Probability Tensors correspond with independencies of (hidden) random variables.
        \end{block}

    \end{frame}



    \begin{frame}{Example: Being at a dentist}

        Add a variable \emph{Cloud}, denoting the weather outside the dentists lab.
        \begin{itemize}
            \item This adds an additional axis to $\probtensor$, thus the number of coordinates increases by a factor of $2$.
            \item But: Intuitively, knowing \emph{Cloud} should not affect the probability of having a cavity, so why shall we care?
        \end{itemize}

        \begin{block}{Independence of Cloud to the other Variables}
            After showing that cavity, catch and toothache are independent of cloud, we do not have to consider cloud any more.
        \end{block}

    \end{frame}



    \begin{frame}{Formal definition of Independencies}

        \begin{definition}[Independence]
    	Given a joint distribution of variables $\catvariableof{0}$ and $\catvariableof{1}$, we say that $\catvariableof{0}$ is independent from $\catvariableof{1}$ if for any values $\catindexof{\catvariableof{0}},\catindexof{\catvariableof{1}}$ we have
        \begin{align*}
            \probat{\indexedcatvariableof{0},\indexedcatvariableof{1}}
            = \probat{\indexedcatvariableof{0}}
            \cdot
            \probat{\indexedcatvariableof{1}} \, .
        \end{align*}
        \end{definition}

    \end{frame}


    \begin{frame}{Contraction criterion for independence}

        \begin{theorem}\label{the:independenceProductCriterion}
            Given a probability distribution $\probtensor$, $\catvariableof{0}$ is independent from $\catvariableof{1}$, if and only if
            \begin{align*}
                \probat{\catvariableof{0},\catvariableof{1}}
                = \contractionof{\probtensor}{\catvariableof{0}} \otimes  \contractionof{\probtensor}{\catvariableof{1}} \, .
            \end{align*}
        \end{theorem}

    \end{frame}

        Independence allows the decomposition into

    \section{Conditional independence}

    \begin{frame}{Assumption of independence}

        Indpendence is a very restrictive assumption, corresponding with tensor products being the most "extreme" tensor decomposition.

            {\bf Example:} In the dentist example, no variables are independent from each other.

        Are there more common properties corresponding to more generic tensor decompositions?
    \end{frame}


    \begin{frame}{Formal definition of Conditional Independencies}

        \begin{definition}[Conditional Independence]
	        Given a joint distribution of variables $\catvariableof{0}$, $\catvariableof{1}$ and $\catvariableof{2}$, we say $\catvariableof{0}$ is independent from $\catvariableof{1}$ conditioned on $\catvariableof{2}$ if for any index tuple $(\catindexof{0},\catindexof{1},\catindexof{2})$
            \begin{align*}
                \condprobof{\indexedcatvariableof{0},\indexedcatvariableof{1}}{\indexedcatvariableof{2}}
                =  \, & \condprobof{\indexedcatvariableof{0}}{\indexedcatvariableof{2}}\\
                & \cdot \condprobof{\indexedcatvariableof{1}}{\indexedcatvariableof{2}}   \, .
            \end{align*}
        \end{definition}


    \end{frame}

    \section{Decompositions given independence}

    \begin{frame}{Chain Rule: Decomposing Probabilities}

        \begin{theorem}[Chain Rule]\label{the:chainRule}
            For any joint probability distribution $\probwith$ we have
            \begin{align*}
                \probwith
                = \contractionof{\{
                    \condprobat{\catvariableof{\atomenumerator}}{\catvariableof{[\catenumerator]}} \wcols \catenumeratorin
                    \}
                }{\shortcatvariables} \, .
            \end{align*}
%where we denote for $\catenumerator=0$ $\probat{\catvariableof{0}}=\condprobat{\catvariableof{[}}{}$.
        \end{theorem}
            {\bf Proof sketch:} We apply the Bayes theorem to get
        \begin{align*}
            \probwith
            =\contractionof{\probat{\catvariableof{0}},\condprobat{\catvariableof{1},\cdots,\catvariableof{\catorder-1}}{\catvariableof{0}}}{\shortcatvariables} \, .
        \end{align*}
        and further apply the Bayes theorem iteratively on the conditioned distribution.
    \end{frame}


    \begin{frame}{Markov Chains}

        \begin{theorem}[Markov Chain]\label{the:MarkovChain}
            Let there be a set of variables $\catvariableof{\catenumerator}$ where $\catenumeratorin$.
            When $\catvariableof{\catenumerator}$ is independent of $\catvariableof{[\catenumerator-1]}$ conditioned on $\catvariableof{\catenumerator-1}$ (the Markov Property), then
            \begin{align*}
                \probwith
                = \contractionof{\condprobof{\catvariableof{\catenumerator}}{\catvariableof{\catenumerator-1}}\wcols \catenumeratorin}{
                    \shortcatvariables
                } \, .
            \end{align*}
        \end{theorem}
            {\bf Proof sketch:} We use the chain decomposition and exploit the independence assumptions to get
        \begin{align*}
            \condprobat{\catvariableof{\catenumerator}}{\catvariableof{[\catenumerator]}}
            = \condprobat{\catvariableof{\catenumerator}}{\catvariableof{\catenumerator-1}} \otimes \onesat{\catvariableof{[\catenumerator-1]}} \, .
        \end{align*}

    \end{frame}

    \section{Bayesian networks}

    \begin{frame}{Independence assumptions of Bayesian networks}

        \begin{itemize}
            \item We enumerate the variables by $[\catorder]$.
            \item To each $\catenumeratorin$ we choose a set $\parentsof{\catenumerator}$ of \emph{parent nodes} in $[\catenumerator]=\{0,\cdots,\catenumerator-1\}$ such that
            \begin{align*}
                \condindependent{\catvariableof{\catenumerator}}{\catvariableof{[\catenumerator]\backslash\parentsof{\catenumerator}}}{\catvariableof{\parentsof{\catenumerator}}}
            \end{align*}
        \end{itemize}

            {\bf Example:} Markov chains are the special case of $\parentsof{\catenumerator}=\{\catenumerator-1\}$.
    \end{frame}


    \begin{frame}{Factorization into Bayesian networks}

        \begin{theorem}
            When for each $\catenumeratorin$
            \begin{align*}
                \condindependent{\catvariableof{\catenumerator}}{\catvariableof{[\catenumerator]\backslash\parentsof{\catenumerator}}}{\catvariableof{\parentsof{\catenumerator}}}
            \end{align*}
            then
            \begin{align*}
                \probwith =
                \contractionof{\{\condprobat{\catvariableof{\catenumerator}}{\catvariableof{\parentsof{\catenumerator}}} \wcols \catenumeratorin\}}{
                    \shortcatvariables
                } \, .
            \end{align*}
        \end{theorem}
            {\bf Proof sketch:} As in the proof of the Markov chain factorization theorem, we use the chain factorization and exploit the conditional independences to show that
        \begin{align*}
            \condprobat{\catvariableof{\catenumerator}}{\catvariableof{[\catenumerator]}}
            = \condprobat{\catvariableof{\catenumerator}}{\catvariableof{\parentsof{\catenumerator}}} \otimes \onesat{\catvariableof{[\catenumerator]\backslash\parentsof{\catenumerator}}} \, .
        \end{align*}

    \end{frame}

    \section{Markov networks}


    \begin{frame}{Undirected Graphical Models:\\
    Markov Networks}


        \begin{definition}[Markov Networks]
	Let $\{\hypercoreofat{\edge}{\catvariableof{\edge}} \wcols \edgein\}$ be a Tensor Network on a hypergraph $\graph$.
    The associated Markov Network is the probability distribution of $\{\catvariableof{\node}\,:\,\nodein\}$ defined by
    \[ \probat{\nodevariables} = \frac{
        \contractionof{\{\hypercoreofat{\edge}{\catvariableof{\edge}} \wcols \edgein\}}{\nodevariables}
    }{
        \contractionof{\{\hypercoreofat{\edge}{\catvariableof{\edge}} \wcols \edgein\}}{\varnothing}
    } = \normalizationofwrt{\{\hypercoreofat{\edge}{\catvariableof{\edge}} \wcols \edgein\}}{\nodevariables}{\varnothing} \, . \]
    We call the denominator
    \[\partitionfunctionof{\graph} = \contractionof{\{\hypercoreofat{\edge}{\catvariableof{\edge}} \wcols \edgein\}}{\varnothing} \]
    the \emph{partition function} of the Markov Network.
        \end{definition}

    \end{frame}


    \begin{frame}{Markov network indepencencies}
        
        \begin{theorem}
            A distribution $\probat{\nodevariables}$ has a representation as a Markov network on a hypergraph $\graph$, if and only if for any disjoint subsets $\secnodes,\thirdnodes,\mathcal{I}\subset\nodes$ for which $\mathcal{I}$ separates $\secnodes$ from $\thirdnodes$
            \begin{align*}
                \condindependent{\catvariableof{\secnodes}}{\catvariableof{\thirdnodes}}{\catvariableof{\mathcal{I}}} \, .
            \end{align*}
        \end{theorem}
            {\bf Proof sketch:} This is the Hammersley Clifford factorization theorem, which proof is out of scope in this course.
    \end{frame}
