    \section{Independence of variables}


    \begin{frame}{How to beat the curse of dimensions?}

        Probability distribution of factored systems with states $\facstates$ has
        \[\left(\prod_{\atomenumeratorin}\catdimof{\atomenumerator}\right) -1 \]
        degrees of freedom (coordinates to specify and store).

        \medskip

        {\bf Mitigation:} \emph{Tensor network decompositions}

        \begin{block}{Idea}
            Exploit the "dependence" structure of variables to find decompositions of probability distributions.
        \end{block}

    \end{frame}



    \begin{frame}{Example: At the dentist}

    \begin{frame}{Formal definition of independence}

        \begin{definition}[Independence]
    	Given a joint distribution $\probat{\nodevariables}$ and two disjoint subsets $\nodesa,\nodesb\subset\nodes$, we say that the variables $\catvariableof{\nodesa}$ are independent from the variables $\catvariableof{\nodesb}$ if the marginal distributions $\probat{\catvariableof{\nodesa\cup\nodesb}}$, $\probat{\catvariableof{\nodesa}}$ and $\probat{\catvariableof{\nodesb}}$ satisfy for any index tuples $\catindexof{\nodesa},\catindexof{\nodesb}$
        \begin{align*}
            \probat{\indexedcatvariableof{\nodesa},\indexedcatvariableof{\nodesb}}
            = \probat{\indexedcatvariableof{\nodesa}}
            \cdot
            \probat{\indexedcatvariableof{\nodesb}} \, .
        \end{align*}
        We denote in this case $\independent{\catvariableof{\nodesa}}{\catvariableof{\nodesb}}$.
        \end{definition}

            {\bf Interpretation:} Knowing the value of variables $\nodesb$ (and conditioning on them) does not change the distribution of variables $\nodesa$.

    \end{frame}


    \begin{frame}{Contraction criterion for independence}

        \begin{theorem}\label{the:independenceProductCriterion}
            If and only if the $\catvariableof{\nodesa}$ are independent from the variables $\catvariableof{\nodesb}$  with respect to a distribution $\probat{\nodevariables}$ then:
            \begin{align*}
                \contractionof{\probwith}{\catvariableof{\nodesa},\catvariableof{\nodesb}}
%                \probat{\catvariableof{\nodesa},\catvariableof{\nodesb}}
                = %\probat{\catvariableof{\nodesa}} \otimes \probat{\catvariableof{\nodesb}}
                \contractionof{\probwith}{\catvariableof{\nodesa}}
                \otimes
                \contractionof{\probwith}{\catvariableof{\nodesb}}
                \, .
            \end{align*}
%            Given a probability distribution $\probtensor$, $\catvariableof{0}$ is independent from $\catvariableof{1}$, if and only if

    \section{Conditional independence}




    \begin{frame}{Assumption of independence}

        Independence is a very restrictive assumption:
        \begin{itemize}
            \item We need dependence of effects on causes for machine learning to work.
            \item Independence covers only the most extreme form of tensor network decompositions.
        \end{itemize}
        \medskip
        {\bf Example:} In the dentist example (without the \emph{Cloud} variable), no variables are independent from each other.
        \begin{center}
            \bf Are there more common properties corresponding to more generic tensor decompositions?
        \end{center}
    \end{frame}


    \begin{frame}{Formal definition of Conditional Independencies}

        \begin{definition}[Conditional Independence]
	        Given a joint distribution of variables $\catvariableof{\nodesa}$, $\catvariableof{\nodesb}$ and $\catvariableof{\nodesc}$, we say that $\catvariableof{\nodesa}$ and $\catvariableof{\nodesb}$ are independent given $\catvariableof{\nodesc}$ if for any index tuple $(\catindexof{\nodesa},\catindexof{\nodesb},\catindexof{\nodesc})$
            \begin{align*}
                \condprobof{\indexedcatvariableof{\nodesa},\indexedcatvariableof{\nodesb}}{\indexedcatvariableof{\nodesc}}
                =  \, & \condprobof{\indexedcatvariableof{\nodesa}}{\indexedcatvariableof{\nodesc}}\\
                & \cdot \condprobof{\indexedcatvariableof{\nodesb}}{\indexedcatvariableof{\nodesc}}   \, ,
            \end{align*}
            We denote this situation by
            \begin{align*}
                \condindependent{\catvariableof{\nodesa}}{\catvariableof{\nodesb}}{\catvariableof{\nodesc}} \, .
            \end{align*}
        \end{definition}

            {\bf Interpretation:} Conditional independence means that $\catvariableof{\nodesa}$ and $\catvariableof{\nodesb}$ are independent with respect to any slice $\condprobat{\catvariableof{\nodesa},\catvariableof{\nodesb}}{\indexedcatvariableof{\nodesc}}$.

    \end{frame}


    \begin{frame}{Conditional independence as a contraction equation}

        We have $ \condindependent{\catvariableof{\nodesa}}{\catvariableof{\nodesb}}{\catvariableof{\nodesc}} $ if and only if
        \begin{align*}
            \condprobat{\catvariableof{\nodesa},\catvariableof{\nodesb}}{\catvariableof{\nodesc}}
            = \contractionof{
                \condprobat{\catvariableof{\nodesa}}{\catvariableof{\nodesc}},
                \condprobat{\catvariableof{\nodesb}}{\catvariableof{\nodesc}}
            }{\catvariableof{\nodesa},\catvariableof{\nodesb},\catvariableof{\nodesc}} \, .
        \end{align*}

        \end{center}

    \end{frame}

    \section{Decomposition schemes}

    \begin{frame}{Chain Rule: Decomposing Probabilities}

        \begin{theorem}[Chain Rule]\label{the:chainRule}
            For any joint probability distribution $\probwith$ we have
            \begin{align*}
                \probwith
                = \contractionof{\{
                    \condprobat{\catvariableof{\catenumerator}}{\catvariableof{[\catenumerator]}} \wcols \catenumeratorin
                    \}
                }{\shortcatvariables} \, .
            \end{align*}
        \end{theorem}
            {\bf Proof sketch:} We apply the Bayes theorem to get
        \begin{align*}
            \probwith
            =\contractionof{\probat{\catvariableof{0}},\condprobat{\catvariableof{1},\cdots,\catvariableof{\catorder-1}}{\catvariableof{0}}}{\shortcatvariables} \, .
        \end{align*}
        and further apply the Bayes theorem iteratively on the conditioned distribution.
        \par
        {\bf Problem:}
        The tensors $\condprobat{\catvariableof{\catenumerator}}{\catvariableof{[\catenumerator]}}$ are of order $\catorder+1$!
        \begin{center}
            \bf Can we reduce the order based on independence assumptions?
        \end{center}

    \end{frame}


    \begin{frame}{Simplification of conditional distributions}

        \begin{theorem}
            When $\condindependent{\catvariableof{\nodesa}}{\catvariableof{\nodesb}}{\catvariableof{\nodesc}}$ then % more precise: need to demand that the cond distributions exist!
            \begin{align*}
                \condprobat{\catvariableof{\nodesa}}{\catvariableof{\nodesc},\catvariableof{\nodesb}}
                = \condprobat{\catvariableof{\nodesa}}{\catvariableof{\nodesc}} \otimes \onesat{\catvariableof{\nodesb}} \, .
            \end{align*}
        \end{theorem}

            {\bf Proof sketch:} We have by Bayes theorem
        \begin{align*}
            \condprobat{\catvariableof{\nodesa},\catvariableof{\nodesb}}{\catvariableof{\nodesc}}
            = \contractionof{
                \condprobat{\catvariableof{\nodesa}}{\catvariableof{\nodesc},\catvariableof{\nodesb}},
                \condprobat{\catvariableof{\nodesb}}{\catvariableof{\nodesc}}
            }{\catvariableof{\nodesa},\catvariableof{\nodesb},\catvariableof{\nodesc}}
        \end{align*}
        and compare with the conditional independence equation
        \begin{align*}
            \condprobat{\catvariableof{\nodesa},\catvariableof{\nodesb}}{\catvariableof{\nodesc}}
            = \contractionof{
                \condprobat{\catvariableof{\nodesa}}{\catvariableof{\nodesc}},
                \condprobat{\catvariableof{\nodesb}}{\catvariableof{\nodesc}}
            }{\catvariableof{\nodesa},\catvariableof{\nodesb},\catvariableof{\nodesc}} \, .
        \end{align*}
        \par
        {\bf Usage for decompositions:} We can drop some conditioned variables and effectively reduce the tensor order in the tensors $\condprobat{\catvariableof{\catenumerator}}{\catvariableof{[\catenumerator]}}$.
    \end{frame}


    \begin{frame}{Example: Markov Chains}

        Let us model the weather in a town on all days of a year by variables $\catvariableof{[365]}$ and assume that for each $\catenumerator\in[365]$
        \begin{itemize}
            \item Three states $\catindexof{\catenumerator}=0$ (sunny), $\catindexof{\catenumerator}=1$ (cloudy), $\catindexof{\catenumerator}=2$ (rainy)
            \item The weather on the day $\catenumerator+1$ conditioned on the weather on the day $\catenumerator$ is
            \begin{align*}
                \condprobat{\catvariableof{\catenumerator+1}}{\catvariableof{\catenumerator}}
                = \coloredmatrixof{    0.70 & 0.25 & 0.05 \\
                0.35 & 0.40 & 0.25 \\
                0.10 & 0.40 & 0.50 }{\catvariableof{\catenumerator+1},\catvariableof{\catenumerator}}
            \end{align*}
            \item The weather on the day $\catenumerator+1$ does not depend on the weather before day $\catenumerator$ given the weather before day $\catenumerator$:
            \begin{align*}
                \condindependent{\catvariableof{\catenumerator+1}}{\catvariableof{[\catenumerator]}}{\catvariableof{\catenumerator}}
            \end{align*}
        \end{itemize}
        \only<3->{
            \begin{center}
                \bf This is an example of a Markov chain: Variables depend only on their predecessors.
            \end{center}
        }
    \end{frame}


    \begin{frame}{Example: Markov Chains}

        \begin{theorem}[Markov Chain]\label{the:MarkovChain}
            Let there be a set of variables $\catvariableof{\catenumerator}$ where $\catenumeratorin$.
            When $\catvariableof{\catenumerator}$ is independent of $\catvariableof{[\catenumerator-1]}$ given $\catvariableof{\catenumerator-1}$ (the Markov Property), then
            \begin{align*}
                \probwith
                = \contractionof{\condprobof{\catvariableof{\catenumerator}}{\catvariableof{\catenumerator-1}}\wcols \catenumeratorin}{
                    \shortcatvariables
                } \, .
            \end{align*}
        \end{theorem}
            {\bf Proof sketch:} We use the chain decomposition and exploit the independence assumptions to get
        \begin{align*}
            \condprobat{\catvariableof{\catenumerator}}{\catvariableof{[\catenumerator]}}
            = \condprobat{\catvariableof{\catenumerator}}{\catvariableof{\catenumerator-1}} \otimes \onesat{\catvariableof{[\catenumerator-1]}} \, .
        \end{align*}

    \end{frame}

    \section{Bayesian networks}

    \begin{frame}{Independence assumptions of Bayesian networks}


        Towards decomposing a distribution in a more generic way take the steps:
        \begin{itemize}
            \item We choose an order and enumerate the variables by $[\catorder]$.
            \item To each $\catenumeratorin$ we choose a set $\parentsof{\catenumerator}$ of \emph{parent nodes} in $[\catenumerator]=\{0,\cdots,\catenumerator-1\}$ such that
            \begin{align*}
                \condindependent{\catvariableof{\catenumerator}}{\catvariableof{[\catenumerator]\backslash\parentsof{\catenumerator}}}{\catvariableof{\parentsof{\catenumerator}}}
            \end{align*}
        \end{itemize}

            {\bf Example:} Markov chains are the special case of $\parentsof{\catenumerator}=\{\catenumerator-1\}$.
    \end{frame}


    \begin{frame}{Factorization into Bayesian networks}

        \begin{theorem}
            When for each $\catenumeratorin$
            \begin{align*}
                \condindependent{\catvariableof{\catenumerator}}{\catvariableof{[\catenumerator]\backslash\parentsof{\catenumerator}}}{\catvariableof{\parentsof{\catenumerator}}}
            \end{align*}
            then
            \begin{align*}
                \probwith =
                \contractionof{\{\condprobat{\catvariableof{\catenumerator}}{\catvariableof{\parentsof{\catenumerator}}} \wcols \catenumeratorin\}}{
                    \shortcatvariables
                } \, .
            \end{align*}
        \end{theorem}
            {\bf Proof sketch:} As in the proof of the Markov chain factorization theorem, we use the chain factorization and exploit the conditional independences to show that
        \begin{align*}
            \condprobat{\catvariableof{\catenumerator}}{\catvariableof{[\catenumerator]}}
            = \condprobat{\catvariableof{\catenumerator}}{\catvariableof{\parentsof{\catenumerator}}} \otimes \onesat{\catvariableof{[\catenumerator]\backslash\parentsof{\catenumerator}}} \, .
        \end{align*}

    \end{frame}

        \begin{itemize}
            \item Choose the order $\catvariableof{1},\catvariableof{0},\catvariableof{2}$
            \item For both $\catvariableof{0}$ and $\catvariableof{2}$ we take $\catvariableof{1}$ as single parent, that is
            \begin{align*}
                \parentsof{0} = \{1\} \quad \text{and} \quad \parentsof{2} = \{1\}
            \end{align*}
            and have the for the Bayesian network necessary conditional independencies (shown before)
            \begin{align*}
                \condindependent{\catvariableof{0}}{\catvariableof{2}}{\catvariableof{1}}
                \quad \text{and} \quad \condindependent{\catvariableof{2}}{\catvariableof{0}}{\catvariableof{1}}
            \end{align*}
        \end{itemize}

    \begin{frame}{Undirected Graphical Models:\\
    Markov Networks}


        \begin{definition}[Markov Networks]
	Let $\{\hypercoreofat{\edge}{\catvariableof{\edge}} \wcols \edgein\}$ be a Tensor Network on a hypergraph $\graph$.
    The associated Markov Network is the probability distribution of $\{\catvariableof{\node}\,:\,\nodein\}$ defined by
    \[ \probat{\nodevariables} = \frac{
        \contractionof{\{\hypercoreofat{\edge}{\catvariableof{\edge}} \wcols \edgein\}}{\nodevariables}
    }{
        \contractionof{\{\hypercoreofat{\edge}{\catvariableof{\edge}} \wcols \edgein\}}{\varnothing}
    } = \normalizationofwrt{\{\hypercoreofat{\edge}{\catvariableof{\edge}} \wcols \edgein\}}{\nodevariables}{\varnothing} \, . \]
    We call the denominator
    \[\partitionfunctionof{\graph} = \contractionof{\{\hypercoreofat{\edge}{\catvariableof{\edge}} \wcols \edgein\}}{\varnothing} \]
    the \emph{partition function} of the Markov Network.
        \end{definition}

    \end{frame}


    \begin{frame}{Markov network independencies}

        Which conditional independencies hold in a Markov network on a hypergraph $\graph$?

        \begin{definition}
            For any hypergraph $\graph=(\edges,\nodes)$ we define:
            \begin{itemize}
                \item A \emph{path} through $\graph$ is a selection of nodes $[\catorder]\subset\nodes$ such that for any $\catenumerator\in[\catorder-1]$ we find a hyperedge $\edgein$ with $\catenumerator,\catenumerator+1\in\edge$.
                \item Given disjoint subsets $\nodesa,\nodesb,\nodesc\subset\nodes$ we say that $\nodesc$ \emph{separates} $\nodesa$ from $\nodesb$ if any path starting at a node in $\nodesa$ and ending at a node in $\nodesb$ goes through $\nodesc$.
            \end{itemize}
        \end{definition}

            {\bf Example:} In a Markov chain graph, for any $\catenumeratorin$ any subset $\nodesa\subset[\catenumerator]$ and subset $\nodesb\subset[\catorder]\backslash[\catenumerator+1]$ is separated by $\nodesc=\{\catenumerator\}$.

    \begin{frame}{Markov network indepencencies}

        The other direction is true for positive distributions: We need to assume for any $\shortcatindices$
        \begin{align*}
            \probat{\indexedshortcatvariables}\neq 0 \, .
        \end{align*}

        \begin{theorem}{Hammersley Clifford}
            A positive distribution $\probat{\nodevariables}$ has a representation as a Markov network on a hypergraph $\graph$, if and only if for any disjoint subsets $\nodesa,\nodesb,\nodesc\subset\nodes$ for which $\nodesc$ separates $\nodesa$ from $\nodesb$
            \begin{align*}
                \condindependent{\catvariableof{\nodesa}}{\catvariableof{\nodesb}}{\catvariableof{\nodesc}} \, .
            \end{align*}
        \end{theorem}
            {\bf Proof:} Very technical, see the book [Koller, Friedman: Probabilistic Graphical Models 2009] and alternatively using the contraction formalism the paper [Goessmann et al: A tensor network formalism for neuro-symbolic AI].
    \end{frame}
