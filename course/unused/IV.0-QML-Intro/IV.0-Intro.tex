    \section{Motivation}

    \begin{frame}{Contraction Bottleneck}

        Reasoning requires contractions

        But: Contractions are often hard to compute, thus a computational bottleneck

        Approximative contractions:
        \begin{itemize}
            \item Message passing
            \item Particle based
        \end{itemize}

    \end{frame}

    \section{Quantum Mechanics}

    \begin{frame}{Tensor formulation}

        States of multiple qubits are in tensor spaces! (Central axiom of quantum mechanics)

    \end{frame}

    \begin{frame}{Quantum circuits}

        Quantum circuits are tensor networks

    \end{frame}

    \section{Outlook}

    \begin{frame}{Quantum inference}

        We investigate in this chapter, how to exploit quantum algorithms for inference of \ComputationActivationNetworks{}:
        \begin{itemize}
            \item Computation of means (forward inference)
            \item Draw samples
            \item Decide satisfiability
        \end{itemize}
