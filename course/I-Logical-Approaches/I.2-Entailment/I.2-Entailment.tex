    \section{Knowledge Bases}

    \begin{frame}{Recall: Curse of Dimensionality}

        Number of coordinates of a propositional formula increases exponentially with the number $\catorder$ of variables.

        Its not just a depiction problem - it is a fundamental intractability in storage and processing!

        \begin{block}{Curse of dimensionality}
            When reasoning about a factored system with many variables (i.e. $\atomorder$ is large), the naive enumeration of all models to formulas is infeasible.
        \end{block}

    \end{frame}

    \begin{frame}{Motivation: Knowledge Base Representation}

        There is further sparsity in the
        \begin{itemize}
            \item We often have a list of formulas.
            \item Each formula is just affected by a few variables.
        \end{itemize}

        \begin{definition}{Knowledge base}
            Knowledge bases are conjunctions over sets of formulas $\{\formulaof{\selindex} \wcols \selindexin\}$
            \begin{align*}
                \kb = \bigwedge_{\selindexin} \formulaof{\selindex}
            \end{align*}
        \end{definition}

        \begin{center}
            \bf How to exploit this sparsity in the effective representation of our knowledge?
        \end{center}

    \end{frame}


    \section{Contraction of Tensors}

    \begin{frame}{Separated Combination of Tensors: \\ Tensor Products}
        \begin{definition}[Tensor Product]
            \label{def:tensorProduct}
            Let there be two tensor
            \begin{align*}
                \hypercore \facstates \rightarrow \rr \quad \text{and} \quad  \sechypercore \facstates \rightarrow \rr \, .
            \end{align*}
            Then there tensor product is the map
            \begin{align*}
                \hypercore \otimes \sechypercore :  \left(\facstates\right) \times \left(\secfacstates\right) \rightarrow \rr
            \end{align*}
            defined for $\catindices\in\facstates$ and $\seccatindices\in\secfacstates$ as
            \begin{align*}
                \hypercore \otimes \sechypercore(\catindices,\seccatindices)
                =  \hypercore(\catindices)\cdot \sechypercore(\seccatindices) \, .
            \end{align*}
        \end{definition}

    \end{frame}


    \begin{frame}{Example: One-hot encodings}

        \textbf{Example:} The one-hot encoding $\onehotmapof{(0,2,1)} \in  \bigotimes_{\atomenumerator\in[3]}\rr^4$
        with the coordinates
        \begin{align*}
            \onehotmapof{(0,1,1)}(\catindexof{0},\catindexof{1},\catindexof{2})
            = \begin{cases}
                  1 & \text{  if  } \catindexof{0} = 0, \,\, \catindexof{1} = 1 \text{  and  } \catindexof{2} = 1 \\
                  0 & \text{ else}
            \end{cases}
        \end{align*}
        is the tensor product of
        \begin{align*}
            \onehotmapof{(0)}(\catindexof{0})
            = \begin{cases}
                  1 & \text{  if  } \catindexof{0} = 0 \\
                  0 & \text{ else}
            \end{cases}
            \quad  , \quad
            \onehotmapof{(2)}(\catindexof{1})
            = \begin{cases}
                  1 & \text{  if  } \catindexof{1} = 1 \\
                  0 & \text{ else}
            \end{cases}
        \end{align*}
        and
        \begin{align*}
            \onehotmapof{(1)}(\catindexof{2})
            = \begin{cases}
                  1 & \text{  if  } \catindexof{2} = 1 \\
                  0 & \text{ else}
            \end{cases} \, .
        \end{align*}


    \end{frame}

    \begin{frame}{Associating tensors with hyperedges}

        Let us use the hypergraph formalism to depict tensor networks
        \begin{itemize}

            \item Hyperedges are decorated by tensors
            \[ \edge = \{\node \, : \, \node \in \edge \} \quad \text{represents the tensor} \quad \hypercoreof{\edge} \in \bigotimes_{\node\in\edge} \rr^{\catdimof{\node}}\]
        \end{itemize}

    \begin{frame}{Tensor Networks}

        \begin{definition}[Tensor Network]
            Let $\graph=(\nodes,\edges)$ be a hypergraph (that is edges are arbitrary nonempty subsets of nodes) with nodes decorated by random variables $\catvariableof{\node}$ with dimensions
            \[ \catdimof{\node} \in \mathbb{N} \]
            and hyperedges $\edge\in\edges$ decorated by core tensors
            \[ \hypercoreof{\edge} \in \bigotimes_{\node\in\edge}\rr^{\catdimof{\node}} \, . \]
            Then we call the set
            \[ \tnetof{\graph} = \{\hypercoreof{\edge} \, : \, \edge\in\edges\} \]
            the Tensor Network of the decorated hypergraph $\graph$.
        \end{definition}

    \end{frame}

    \begin{frame}{Contraction of Tensor Networks}

        \begin{definition}[Contraction]
            Let $\tnetof{\graph}$ be a tensor network on a decorated hypergraph $\graph=(\nodes,\edges)$.
            For any subset $\secnodes\subset\nodes$ we define the contraction to be the tensor
            \begin{align}
                \contractionof{\tnetof{\graph}}{\catvariableof{\secnodes}} \in \bigotimes_{\node\in\secnodes} \rr^{\catdimof{\node}}
            \end{align}
            defined coordinatewise by the sum
            \begin{align}
                \contractionof{\tnetof{\graph}}{\indexedcatvariableof{\secnodes}} =
                \sum_{\{\catindexof{\node} \in \catdimof{\node} \, : \, \node \in \nodes/\secnodes\}}
                \prod_{\edge\in\edges}\hypercoreofat{\edge}{\indexedcatvariableof{\edge}} \, .
            \end{align}
        \end{definition}

    \end{frame}

    \begin{frame}{Curse of dimensionality of factored representations}

        Tensor spaces have a problem with exponential increase of dimensions as
        \[ \mathrm{dim}\left[
                           \bigotimes_{\node\in\nodes} \rr^{\catdimof{\node}}
        \right] =
        \prod_{\node\in\nodes} \catdimof{\node} \, . \]


        Tensor Networks can break this curse of dimensionality by storing the decorating tensors instead
        \begin{align*}
            \sum_{\edge\in\edges} \mathrm{dim}\left[  \bigotimes_{\node\in\edge} \rr^{\catdimof{\node}} \right]
            =    \sum_{\edge\in\edges}  \left(\prod_{\node\in\edge} \catdimof{\node}\right)
        \end{align*}


        \begin{block}{Representation strategy}
            Store the recipe (that is the core tensors), not the result (the contracted tensor network).
        \end{block}

    \end{frame}

    \begin{frame}{Knowledge Bases as Contractions}

        We model each formula in a Knowledge base as a decoration
        \begin{itemize}
            \item hyperedge $\edgeof{\selindex}$ of affected variables
            \item decorating tensor $\formulaofat{\selindex}{\catvariableof{\edgeof{\selindex}}}$
        \end{itemize}
        For any Knowledge base we have
        \begin{align*}
            \kbat{\nodevariables}
            = \contractionof{\formulaofat{\selindex}{\catvariableof{\edgeof{\selindex}}}}{\nodevariables}
        \end{align*}
    \end{frame}

    \begin{frame}{Model counts}

        So far: Knowledge bases as contractions with all variables open.
        Now: Contractions without open variables.

        We have
        \begin{align*}
            &\contraction{\kbwith,\formulawith} \\
            &\quad = \#\{\shortcatindices\in[2]^{\catorder} \wcols \kbat{\indexedshortcatvariables}=\formulaat{\indexedshortcatvariables}=1\}
        \end{align*}
        That is, the number of common models is counted

        So if
        \begin{align*}
            \contraction{\kbwith,\formulawith} = \contraction{\kbwith}
        \end{align*}
        then $\kb\models\formula$.


    \end{frame}

    \begin{frame}{Entailment}

        \begin{definition}[Entailment]
            Let $\kbat{\shortcatvariables}$ and $\formulawith$ be two propositional formulas.
            If for all worlds $\shortcatindices$, for which $\kbat{\indexedshortcatvariables}=1$, we also have $\formulaat{\indexedshortcatvariables}=1$, then we say that $\formula$ is entailed by $\kb$ and denote
            \begin{align*}
                \kb \models \formula
            \end{align*}

        \end{definition}

        \textbf{Aim:} How to express this as a computation?

        \begin{center}
            \bf We develop the contraction formalism of tensors for this.
        \end{center}

    \end{frame}


    \begin{frame}{Entailment by contractions}

        Trick:
        \begin{align*}
            \contraction{\kbwith,\formulawith} = \contraction{\kbwith}
        \end{align*}
        is equal to
        \begin{align*}
            \contraction{\kbwith}-\contraction{\kbwith,\formulawith} = 0
        \end{align*}
        and
        \begin{align*}
            \contraction{\kbwith,\lnot\formulawith} = 0
        \end{align*}

        \begin{block}{Contraction criterion for entailment}
            For any propositional formulas we have $\kb\models\formula$ if and only if
            \begin{align*}
                \contraction{\kbwith,\lnot\formulawith} = 0
            \end{align*}
        \end{block}
    \end{frame}

    \begin{frame}{Subpackage \spengine{}: Cores}

        Each Tensor core has attributes
        \begin{itemize}
            \item \emph{values} (array-like): storing the value of the coordinates
            \item \emph{colors} (list of str): specifying the name of the variables represented by its axes
            \item \emph{name} (str): to distinguish from other cores
        \end{itemize}
        The implemented core types differ in the values argument.
        Cores are instantiated by
        \begin{centeredscript}
            \emph{engine.getCore(coreType)(coreValues, coreColors, coreName)}
        \end{centeredscript}

    \end{frame}


    \begin{frame}{Subpackage \spengine{}: Contractions}

        Reflected in the notation
        \[ \contractionof{\tnetof{\graph}}{\nodes} \]
        a contraction is defined by
        \begin{itemize}
            \item Tensor Network $\tnetof{\graph}$, i.e. a dictionary of tensor cores
            \item Open Variables $\nodes$
        \end{itemize}
        Contraction calls are done by
        \begin{centeredscript}
            \emph{engine.contract(contractionMethod, coreDict, openColors)}
        \end{centeredscript}
        Where
        \begin{itemize}
            \item \emph{contractionMethod:} str, chooses one of the contraction providers
            \item \emph{coreDict}: Dictionary of TensorCores (of the above formats), representing the Tensor Network $\tnetof{\graph}$
            \item \emph{openColors}: List $\nodes$ of str, each str identifying a color, that is a variable to be left open in the contraction
        \end{itemize}
    \end{frame}
