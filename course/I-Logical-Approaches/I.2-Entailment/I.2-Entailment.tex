    \section{Knowledge Bases}

    \begin{frame}{Motivation: Curse of dimensionality}
        \begin{block}{Observation}
            The number of coordinates of a propositional formula increases exponentially with the number $\catorder$ of variables.
        \end{block}
        \begin{center}
            \bf  Its not just a depiction problem - it is a fundamental intractability in storage and processing!
        \end{center}

        \begin{block}{Curse of dimensionality: Storage problem}
            When reasoning about a factored system with many variables (i.e. $\atomorder$ is large), the naive enumeration of all models to formulas is infeasible.
        \end{block}
    \end{frame}

    \begin{frame}{Motivation: Knowledge bases}
        There are further sparsity principles:
        \begin{itemize}
            \item We often have a conjunction over multiple formulas.
            \item Each formula is just affected by a few variables.
        \end{itemize}

        \begin{definition}[Knowledge base]
            Knowledge bases are conjunctions over formulas $\{\formulaof{\selindex} \wcols \selindexin\}$, that is
            \begin{align*}
                \kb = \bigwedge_{\selindexin} \formulaof{\selindex} \, .
            \end{align*}
            We represent them by a tensor
            \begin{align*}
                \kbat{\indexedshortcatvariables}
                = \begin{cases}
                      1 & \ifspace \uniquantwrtof{\selindexin}{\formulaofat{\selindex}{\indexedshortcatvariables} = 1} \\
                      0 & \text{else} \, .
                \end{cases}
            \end{align*}
        \end{definition}

        \begin{center}
            \bf How to exploit this sparsity for more effective representation?
        \end{center}

    \end{frame}

    \begin{frame}{Tensor Networks}
        \begin{definition}[Tensor Network]
            Let $\graph=(\nodes,\edges)$ be a hypergraph with nodes decorated by variables $\catvariableof{\node}$ with dimensions $\catdimof{\node} \in \mathbb{N}$.
            and hyperedges $\edge\in\edges$ decorated by core tensors
            \[ \hypercoreofat{\edge}{\catvariableof{\edge}} \in \bigotimes_{\node\in\edge}\rr^{\catdimof{\node}} \, . \]
            Then the set $\{\hypercoreofat{\edge}{\catvariableof{\edge}}\wcols\edgein\}$ builds the \emph{tensor network} to the decorated hypergraph $\graph$ and we say it decomposes the tensor
            \[ \tnetofat{\graph}{\nodevariables} \coloneqq
            \contractionof{\hypercoreofat{\edge}{\catvariableof{\edge}} \wcols \edge\in\edges}{\nodevariables} \, .
            \]
            with coordinates to $\catindexof{\nodes}\in\bigtimes_{\nodein}[\catdimof{\node}]$ by
            \begin{align*}
                \tnetofat{\graph}{\indexedcatvariableof{\nodes}}
                = \prod_{\edgein} \hypercoreofat{\edge}{\indexedcatvariableof{\edge}} \, .
            \end{align*}
        \end{definition}
    \end{frame}

    \begin{frame}{Curse of dimensionality of factored representations}
        Tensor spaces have a problem with exponential increase of dimensions as
        \[ \mathrm{dim}\left[
                           \bigotimes_{\node\in\nodes} \rr^{\catdimof{\node}}
        \right] =
        \prod_{\node\in\nodes} \catdimof{\node} \, . \]

        Tensor Networks can break this curse of dimensionality by storing the decorating tensors instead
        \begin{align*}
            \sum_{\edge\in\edges} \mathrm{dim}\left[  \bigotimes_{\node\in\edge} \rr^{\catdimof{\node}} \right]
            =    \sum_{\edge\in\edges}  \left(\prod_{\node\in\edge} \catdimof{\node}\right)
        \end{align*}

        \begin{block}{Representation strategy}
            Store the recipe (that is the core tensors), not the result (the contracted tensor network).
        \end{block}
    \end{frame}

    \begin{frame}{Knowledge bases as tensor networks}
        \only<1>{
            Given a knowledge base $\kb=\{\formulaof{\selindex}\wcols\selindexin\}$ we
            \begin{itemize}
                \item Create for each $\selindexin$ a hyperedge $\edgeof{\selindex}$ containing the nodes to the variables $\catvariableof{\edgeof{\selindex}}$ appearing in the formula $\formulaof{\selindex}$
                \item Build a set of nodes, which correspond with the variables appearing in the formulas
                \begin{align*}
                    \nodes = \bigcup_{\selindexin}\edgeof{\selindex}
                \end{align*}
            \end{itemize}
        }
        \begin{theorem}[Tensor network decomposition of knowledge bases]
            For any knowledge base $\kb=\{\formulaof{\selindex}\wcols\selindexin\}$ the corresponding tensor $\kbat{\nodevariables}$ is decomposed into the formulas by
            \begin{align*}
                \kbat{\nodevariables}
                = \contractionof{\{\formulaofat{\selindex}{\catvariableof{\edgeof{\selindex}}}\wcols\selindexin\}}{\nodevariables} \, .
            \end{align*}
        \end{theorem}
        \only<2>{
            \textbf{Proof:}
            For each $\catindexof{\nodes}\in\bigtimes_{\node\in\nodes}[2]$ we have $\kbat{\indexedcatvariableof{\nodes}}=1$ if and only if for all $\selindexin$ $\formulaofat{\selindex}{\indexedcatvariableof{\edgeof{\selindex}}}=1$.
            This is equal to
            \begin{align*}
                \kbat{\indexedcatvariableof{\nodes}}
                = \prod_{\selindexin} \formulaofat{\selindex}{\indexedcatvariableof{\edgeof{\selindex}}} \, .
            \end{align*}
        }

    \end{frame}

    \begin{frame}{Scalar products of vectors}
        Let there be a variable $\catvariable$ taking values in $[\catdim]$ and vectors
        \begin{align*}
            \vectorofat{0}{\catvariable}\in\rr^{\catdim} \quad \text{and}\quad\vectorofat{1}{\catvariable}\in\rr^{\catdim} \, .
        \end{align*}
        We denote their scalar product by % $\langle\vectorof{0},\vectorof{1}\rangle$ is
        \begin{align*}
            \contraction{\vectorofat{0}{\catvariable},\vectorofat{1}{\catvariable}}
            = \sum_{\catindexin} \vectorofat{0}{\indexedcatvariable} \cdot \vectorofat{1}{\indexedcatvariable} \, .
        \end{align*}
        and visualize it diagramatically with \emph{closed legs}:
        \stantikzpic{
            \drawsmalltensor{0}{0}{\corelabelsize $\vectorof{0}$}
            \draw (1,0) -- (3,0) node[midway,above] {\colorlabelsize $\catvariable$};
            \drawsmalltensor{4}{0}{\corelabelsize $\vectorof{1}$}
        }
        \begin{block}{Einstein sum convention}
            The indices to variables, which not left open in an contraction, are summed over.
        \end{block}
    \end{frame}

    \begin{frame}{Contraction of Tensor Networks}
        \begin{definition}[Contraction]
            Let $\tnetof{\graph}$ be a tensor network on a decorated hypergraph $\graph=(\nodes,\edges)$.
            For any subset $\secnodes\subset\nodes$ we define the contraction to be the tensor
            \begin{align*}
                \contractionof{\tnetof{\graph}}{\catvariableof{\secnodes}} \in \bigotimes_{\node\in\secnodes} \rr^{\catdimof{\node}}
            \end{align*}
            defined coordinatewise by the sum
            \begin{align*}
                \contractionof{\tnetof{\graph}}{\indexedcatvariableof{\secnodes}} =
                \sum_{\catindexof{\nodes\backslash\secnodes}\in\bigtimes_{\node\in\nodes\backslash\secnodes}[\catdimof{\node}]}
                \prod_{\edge\in\edges}\hypercoreofat{\edge}{\indexedcatvariableof{\edge}} \, .
            \end{align*}
        \end{definition}
    \end{frame}

    \section{Entailment}

    \begin{frame}{Contraction without open variables: Model counts}
        \par
        {\bf So far:} Knowledge bases are contractions of formulas keeping all variables open.
        \medskip \\
        {\bf Now:} Contractions without open variables:
        \begin{align*}
            \contraction{\formulawith}
            = \sum_{\shortcatindices\in[2]^{\catorder}} \formulaat{\indexedshortcatvariables}
            = \#\{\shortcatindices \wcols \formulaat{\indexedshortcatvariables}=1\}
        \end{align*}
        \stantikzpic{
            \draw (-1,-1) rectangle (5,1);
            \drawtextnode{2}{0}{\corelabelsize $\exformula$}
            \draw (0,-1) -- (0,-2) node[midway,left] {\colorlabelsize $\catvariableof{0}$};
            \drawvariabledot{0}{-2}
            \drawtextnode{2}{-1.5}{$\cdots$}
            \draw (4,-1) -- (4,-2) node[midway,left] {\colorlabelsize $\catvariableof{\catorder-1}$};
            \drawvariabledot{4}{-2}
        }

        \begin{block}{Contraction of formulas without open variables}
            The contraction $\contraction{\formulawith}$ of a formula without leaving variables open is the \emph{count of its models} (that are states $\shortcatindices$ with $\formulaat{\indexedshortcatvariables}=1$).
            If $\contraction{\formulawith}=0$, we say the formula is \emph{unsatisfiable}.
        \end{block}
    \end{frame}

    \begin{frame}{Entailment}
        \begin{definition}[Entailment]
            Let $\kbat{\shortcatvariables}$ and $\formulawith$ be two propositional formulas.
            If for all worlds $\shortcatindices$, for which $\kbat{\indexedshortcatvariables}=1$, we also have $\formulaat{\indexedshortcatvariables}=1$, then we say that $\formula$ is entailed by $\kb$ and denote
            \begin{align*}
                \kb \models \formula \, .
            \end{align*}
        \end{definition}

        \textbf{How to express this as a contraction?}\\
        For any propositional formulas we have $\kb\models\formula$ if and only if
        \begin{align*}
            \contraction{\kbwith,\formulawith}
            = \contraction{\kbwith}
        \end{align*}
    \end{frame}

    \begin{frame}{Vanishing contraction criterion}
        Since $\kb\models\formula$ is equal to
        \begin{align*}
            \contraction{\kbwith,\formulawith} = \contraction{\kbwith}
        \end{align*}
        it is further equal to
        \begin{align*}
            0 =& \contraction{\kbwith}-\contraction{\kbwith,\formulawith} \\
            =& \contraction{\kbwith,\oneswith}-\contraction{\kbwith,\formulawith} \\
            = & \contraction{\kbwith,\oneswith-\formulawith} \\
            =&\contraction{\kbwith,\lnot\formulawith} \, .
        \end{align*}
        \vspace{-0.5cm}
        \begin{block}{Contraction criterion for entailment}
            For any propositional formulas we have $\kb\models\formula$ if and only if
            \begin{align*}
                \contraction{\kbwith,\lnot\formulawith} = 0 \, ,
            \end{align*}
            that is when $\kb\land(\lnot\formula)$ is unsatisfiable.
        \end{block}
    \end{frame}

    \begin{frame}[fragile]{Subpackage \spengine{}: Tensors}
        The class \pyinline{engine.TensorCore} has attributes
        \begin{itemize}
            \item \emph{colors} (list of str): stores strings representing $\catvariableof{\catenumerator}$ for $\catenumeratorin$
            \item \emph{shape} (list of int): stores the leg dimensions $\catdimof{\catenumerator}$ for $\catenumeratorin$
            \item \emph{name} (str): to distinguish from other cores
            \item \emph{values} (array-like): stores the coordinates (differing in sub-classes)
        \end{itemize}
        The implemented core types differ in the values argument.
        Cores are instantiated by
        \begin{minted}{python}
    from tnreason import engine
    tensorCore = engine.getCore(coreType)(coreValues, coreColors, coreName)
        \end{minted}
    \end{frame}

    \begin{frame}[fragile]{Subpackage \spengine{}: Tensor networks}
        Tensor networks $\tnetof{\graph}$ in \tnreason{} are dictionaries \pyinline{coreDict} of str tensor nemes valued by \pyinline{engine.TensorCore} instances.
        \medskip \\
        The hypergraph $\graph=(\edges,\nodes)$ is stored
        \begin{itemize}
            \item Each $\edge$ is represented by an instance \pyinline{tensorCore} of \pyinline{engine.TensorCore}
            \item The incidence relations are stored by \pyinline{tensorCore.values}
        \end{itemize}
        \medskip
        Visualization with \pyinline{matplotlib}:
        \begin{minted}{python}
    engine.draw_fractor_graph(coreDict)
        \end{minted}
    \end{frame}

    \begin{frame}[fragile]{Subpackage \spengine{}: Contractions}
        Reflected in the notation
        \begin{align*}
            \contractionof{\tnetof{\graph}}{\catvariableof{\secnodes}}
        \end{align*}
        contraction calls are done by
        \begin{minted}{python}
    from tnreason import engine
    contractedNetwork = engine.contract(coreDict, openColors)
        \end{minted}
        Where
        \begin{itemize}
            \item \emph{coreDict}: Dictionary of tensor cores representing $\extnet$
            \item \emph{openColors}: List of str identifying the variables (and their order) $\catvariableof{\secnodes}$ left open in the contraction
        \end{itemize}
    \end{frame}
