    \begin{frame}{Circuits}

        \begin{definition}[Circuit]
            A circuit is a decorated hypergraph $\graph=(\nodes,\edges)$, where:
            \begin{itemize}
                \item Each node $\node\in\nodes$ is decorated by a dimension $\catdimof{\node}\in\nn$
                \item Each edge $\edge\in\edges$ is a tuple of incoming nodes $\incomingnodes$ and outgoing nodes $\outgoingnodes$
                \item Each edge $\edge\in\edges$ is decorated by a function
                \begin{align*}
                    \exfunctionof{\edge}
                    \defcols \bigtimes_{\node\in\incomingnodes}[\catdimof{\node}] \rightarrow \bigtimes_{\node\in\outgoingnodes}[\catdimof{\node}]
                \end{align*}
            \end{itemize}

            We assume that:
            \begin{itemize}
                \item The hypergraph is acyclic
                \item Each node appears at most once as an outgoing node
            \end{itemize}


        \end{definition}

        %Propositional syntax decomposes formulas into connectives applied on sub-formulas.

%        When understanding formulas as functions, we get a decomposition into circuits of connectives.

        %  This decomposition is reflected in the tensor network decomposition of basis encodings.


    \end{frame}


    \begin{frame}{Functions represented by circuits}

        We denote by
        \begin{itemize}
            \item $\innodes$ the nodes appearing never as outgoing nodes.
            \item $\outnodes$ the nodes appearing never as incoming nodes.
        \end{itemize}

        Now we can iteratively define a function
        \begin{align*}
            \secexfunctionof{\node} \defcols \bigtimes_{\secnode\in\incomingnodes}[\catdimof{\secnode}] \rightarrow [\catdimof{\node}]
        \end{align*}
        as the composition of the functions decorating the edges, mapping the states of $\innodes$ to the states of $\outnodes$, by
        \begin{align*}
            \secexfunctionof{\node}  = \exfunctionof{\edge}|_{\node} \circ (\bigtimes_{\secnode\in\incomingnodes} \secexfunctionof{\secnode}) \, .
        \end{align*}


        The circuit represents the function
        \begin{align*}
            \secexfunctionof{\graph}
            = \bigtimes_{\node\in\outnodes} \secexfunctionof{\node} \, .
        \end{align*}
    \end{frame}

    \begin{frame}{Example: Propositional syntax}

        Consider the formula $(\catvariableof{B}\lor\catvariableof{E})\Rightarrow (\catvariableof{J}\lor\catvariableof{M})$
        \begin{itemize}
            \item The nodes are the variables $\catvariableof{B},\catvariableof{E},\catvariableof{J},\catvariableof{M}$, the sub-formulas $\headvariableof{B\lor E},\headvariableof{J\lor M}$, and the formula $\headvariableof{\formula}$
            \item The nodes $\innodes=\{\catvariableof{B},\catvariableof{E},\catvariableof{J},\catvariableof{M}\}$ do not appear as outgoing nodes
            \item The node $\headvariableof{\formula}$ is the only one not appearing as incoming nodes, thus $\outnodes=\{\headvariableof{\formula}\}$
        \end{itemize}

    \begin{frame}{General basis encodings}


        Consider an arbitrary function
        \begin{align*}
            \secexfunction \defcols \bigtimes_{\node\in\innodes}[\catdimof{\node}] \rightarrow \bigtimes_{\node\in\outnodes}[\catdimof{\node}] \, .
        \end{align*}
        We define its \emph{basis encoding} as the tensor
        \begin{align*}
            \bencodingofat{\secexfunction}{
                \catvariableof{\outnodes},\catvariableof{\innodes}
            }
            = \sum_{\catindexof{\innodes}\in\bigtimes_{\node\in\innodes}[\catdimof{\node}]}
            \onehotmapofat{\secexfunction(\catindexof{\innodes})}{\catvariableof{\outnodes}}
            \otimes
            \onehotmapofat{\catindexof{\innodes}}{\catvariableof{\innodes}} \, .
        \end{align*}

    \begin{frame}{Decomposition into tensor networks}
        To each edge $\edge$ we choose a basis encoding
        \begin{align*}
            \bencodingofat{\exfunctionof{\edge}}{\catvariableof{\outgoingnodes},\catvariableof{\incomingnodes}}
            = \sum_{\catindexof{\incomingnodes}\in \bigtimes_{\node\in\incomingnodes}[\catdimof{\node}]}
            \onehotmapofat{\exfunctionof{\edge}(\catindexof{\incomingnodes})}{\catvariableof{\outgoingnodes}}
            \otimes \onehotmapofat{\catindexof{\incomingnodes}}{\catvariableof{\incomingnodes}}
        \end{align*}
        The basis encoding to a function of a circuit is decomposed into basis encodings of the node functions:
        \begin{align*}
            \bencodingofat{\exfunctionof{\graph}}{
                \catvariableof{\outnodes},\catvariableof{\innodes}
            }
            = \contractionof{
                \{\bencodingofat{\exfunctionof{\edge}}{\catvariableof{\outgoingnodes},\catvariableof{\incomingnodes}} \wcols \edge\in\edges\}
            }{\catvariableof{\outnodes},\catvariableof{\innodes}}
        \end{align*}

    \section{Example: $\catdim$-adic sums}


    \begin{frame}{$\catdim$-adic representation of integers}
        Let $\catdim,\catorder$ be arbitrary natural numbers.
        Any number $n\in[\catdim^{\catorder}]$ has a unique $\catdim$-adic representation
        \begin{align*}
            \sum_{\catenumeratorin} \catindexof{\catenumerator} \cdot \catdim^{\catenumerator}
        \end{align*}
        where for each $\catenumeratorin$
        \begin{align*}
            \catindexof{\catenumerator} \in [\catdim] \, .
        \end{align*}

            {\bf Example:} For $\catdim=3$ and $\catorder=2$, the number $5\in[9]$ has the $3$-adic representation
        \begin{align*}
            (\catindexof{0},\catindexof{1}) = (2,1)
        \end{align*}
        since
        \begin{align*}
            5 = 2\cdot 3^0 + 1\cdot 3^1 \, .
        \end{align*}

    \end{frame}


    \begin{frame}{Summation function}
        We want to find a circuit decomposing the summation function
        \begin{align*}
            \marysumsymbol
            \defcols [\catdim^{\catorder}] \times [\catdim^{\catorder}] \rightarrow [\catdim^{\catorder+1}]
        \end{align*}
        defined for tuples $(\shortcatindices,\tildecatindexof{[\catorder]})\in[\catdim]^{\catorder}$ with $\marysumsymbol(\shortcatindices,\tildecatindexof{[\catorder]})=\headindexof{[\catorder+1]}$ as
        \begin{align*}
            \sum_{\catenumerator\in[\catorder+1]} \headindexof{\catenumerator} \cdot \catdim^{\catenumerator}
            =             \sum_{\catenumeratorin} (\catindexof{\catenumerator} + \tildecatindexof{\catenumerator}) \cdot \catdim^{\catenumerator}\, .
        \end{align*}
        % \end{frame}

    \begin{frame}{Circuit of half-adders}

        Nodes representing the bits to numbers:
        \begin{itemize}
            \item $\shortcatvariables$ representing the first number to be summed
            \item $\tildecatvariableof{[\catorder]}$ representing the second number to be summed
            \item $\headvariableof{[\catorder+1]}$ representing the sum
            \item $\thirdcatvariableof{[\catorder-1]}$ representing carry bits
        \end{itemize}

        Edges are enumerated by $[\catorder-1]=\{0,\ldots,\catorder-2\}$ representing half-adders:
        \begin{align*}
            \exfunctionof{\edgeof{\catenumerator}}(\thirdcatindexof{\catenumerator-1},\catindexof{\catenumerator},\tildecatindexof{\catenumerator})
            = \left(
                  (\thirdcatindexof{\catenumerator-1} + \catindexof{\catenumerator} + \tildecatindexof{\catenumerator}) \modspace\catdim,
                  \left\lfloor\frac{\thirdcatindexof{\catenumerator-1} + \catindexof{\catenumerator} + \tildecatindexof{\catenumerator}}{\catdim}\right\rfloor
            \right)
        \end{align*}
