    \begin{frame}{Propositional Syntax}

        Propositional syntax (e.g. $\formula = \left(\catvariableof{B} \lor \catvariableof{E}\right) \Rightarrow \left(\catvariableof{J} \lor \catvariableof{M}\right)$) consists of
        \begin{itemize}
            \item Atomic formulas (here: $\catvariableof{B}$, $\catvariableof{E}$, $\catvariableof{J}$, $\catvariableof{M}$)
            \item Connectives (here: $\lor$, $\Rightarrow$)
        \end{itemize}
        \input{tikz/syntax-explanation}
        \only<1>{
                {\bf Atomic formulas} ("propositional symbols"), e.g. $\catvariableof{B}$:\\
            A world $(\catindexof{B}$, $\catindexof{E}$, $\catindexof{J}$, $\catindexof{M})\in[2]^4$ is a model, if and only if $\catindexof{B}=1$.
        }
        \only<2->{
            Connectives building {\bf complex formulas}, e.g. $\catvariableof{B}\lor\catvariableof{E}$:\\
            Satisfaction computed by the satisfaction of the sub-formulas
        }

        }

    \end{frame}


    \begin{frame}{How to represent formulas as tensor networks?}

        \begin{block}{Task: Exploit the syntax}
            The syntactical representation of formulas is the sparsity principle of propositional logics to avoid the curse of dimensions.
            Can we capture this principle by tensor network decompositions?
        \end{block}
        To be more precise:
        \begin{itemize}
            \item Suppose we have a tensor network decomposition of formulas $\formulaof{0}$ and $\formulaof{1}$.
            \item How to contract the tensor networks to get $\formulaof{0}\exconnective\formulaof{1}$ (where $\exconnective\in\{\lor,\land,\Rightarrow,\cdots\}$)?
        \end{itemize}
        \medskip
        {\bf Recall:} Contraction of both formulas (or tensor network decomposition of them) is the logical conjunction
        \begin{align*}
            \Big(\formulaof{0}\land\formulaof{1}\Big)[\shortcatvariables] =
            \contractionof{\formulaofat{0}{\shortcatvariables},\formulaofat{1}{\shortcatvariables}}{\shortcatvariables}
        \end{align*}
    \end{frame}

    \begin{frame}{The non-linearity problem}
        \begin{block}{Multilinearity of tensor networks}
            Tensor networks are multilinear, that is linear in each core tensor decorating an edge.
        \end{block}

        The logical conjunction is the only connective which is multilinear.

        \begin{block}{Non-linearity problem}
            How to represent non-linear formulas like $\lnot\catvariable$ in a tensor network?
        \end{block}

            {\bf Solution:} \\
        We develop the \emph{basis calculus} scheme instead of \emph{coordinate calculus}.
        % \end{center}

    \end{frame}

    \begin{frame}{Coordinate and basis encoding}

        {\bf Coordinate encoding:} \\
        Store an information $\formulaev{\shortcatindices}$ in the coordinates of a tensor
        \begin{align*}
            \formulaat{\shortcatvariables}
            = \sum_{\shortcatindices\in[2]^{\catorder}}\formulaev{\shortcatindices} \cdot \onehotmapofat{\shortcatindices}{\shortcatvariables}
        \end{align*}

            {\bf Basis encoding:} \\
        Store an information $\formulaev{\shortcatindices}$ in a basis vector $\onehotmapofat{\formulaev{\shortcatindices}}{\headvariableof{\formula}}$ to an auxiliary variable $\headvariableof{\formula}$, that is
        \begin{align*}
            \bencodingofat{\formula}{\headvariableof{\formula},\shortcatvariables}
            = \sum_{\shortcatindices\in[2]^{\catorder}}  \onehotmapofat{\formulaev{\shortcatindices}}{\shortcatvariables} \otimes \onehotmapofat{\shortcatindices}{\shortcatvariables}
        \end{align*}

    \end{frame}

    \begin{frame}{Basis calculus}
        Idea of basis calculus:
        \begin{itemize}
            \item Represent input $\shortcatindices$ by one-hot encoding $\onehotmapofat{\shortcatindices}{\shortcatvariables}$
            \item Contract with the basis encoding of a function
            \item Get one-hot encoding of $\formula(\shortcatindices)$ as a result
        \end{itemize}

        \begin{block}{Basis calculus scheme}
            States are represented by basis vectors (their one-hot encodings) and processed by contractions with basis encodings.
        \end{block}
    \end{frame}

    \begin{frame}{\DecompositionSparsity{}}

        \begin{theorem}
            The basis encoding of a composed formula is the contraction of the basis encodings to its connectives.
        \end{theorem}
        \textbf{Proof sketch:} When contracting the decomposition with one-hot encodings of any $\shortcatindices\in[2]^{\catorder}$, we get by basis calculus the one-hot encodings of the composition formulas, and finally the one-hot encoding of the whole formula.
        This coincides with contracting the full basis encoding since
        \begin{align*}
            \contractionof{\bencodingofat{\formula}{\headvariableof{\exformula},\shortcatvariables},\onehotmapofat{\shortcatindices}{\shortcatvariables}}{
                \headvariableof{\exformula}
            }
            = \onehotmapofat{\formula(\shortcatindices)}{\headvariableof{\formula}} \, .
        \end{align*}

        \begin{block}{\DecompositionSparsity{}}
            When a function can be decomposed into connective functions, its corresponding tensor has a tensor network decomposition by the basis encodings of the connective functions.
%            When there is a syntactical representation of a formula, we find a tensor network decomposition of the formula.
        \end{block}
    \end{frame}

    \section{Rules of replacement and clauses}

    \begin{frame}{Manipulations of syntax}

        There are different syntactical representations of the same formula, some of them are found by syntactical manipulations.
        \begin{align*}
            \textbf{Example: } \quad \text{De Morgan rule} \quad
            \lnot\Big(\catvariableof{0} \land \catvariableof{1}\Big)
            \equiv \Big(\lnot\catvariableof{0} \lor \lnot\catvariableof{1})\Big)
        \end{align*}

        \begin{block}{Tensor network perspective}
            Equivalence of syntactical representations correspond with equivalence of corresponding tensor network contractions.
        \end{block}

        The De Morgan rule corresponds with the contraction equation:
        \input{tikz/de-morgan-contraction}

    \end{frame}

    \begin{frame}{Rules of replacement}
        For any propositional formulas $\formulaof{0}$ and $\formulaof{1}$ the following is true:
        \begin{itemize}
            \item Elimination of bi-implications
            \begin{align*}
                \Big(\formulaof{0} \Leftrightarrow \formulaof{1}\Big)
                \equiv
                \Big((\formulaof{0}\Rightarrow\formulaof{1}) \land (\formulaof{1}\Rightarrow\formulaof{0})\Big)
            \end{align*}
            \item Elimination of implications
            \begin{align*}
                \Big(\formulaof{0} \Rightarrow \formulaof{1}\Big)
                \equiv
                \Big(\lnot\formulaof{0} \lor \formulaof{1})
            \end{align*}
            \item De Morgan rules
            \begin{align*}
                &\lnot\Big(\formulaof{0} \land \formulaof{1}\Big)
                \equiv
                \Big(\lnot\formulaof{0} \lor \lnot \formulaof{1} \Big)
                \quad, \quad\\
                &\lnot\Big(\formulaof{0} \lor \formulaof{1}\Big)
                \equiv
                \Big(\lnot\formulaof{0} \land \lnot \formulaof{1} \Big)
            \end{align*}
            \item Distributive rule
            \begin{align*}
                &\Big(\formulaof{0} \lor (\formulaof{1} \land \formulaof{2})\Big)
                \equiv
                \Big((\formulaof{0} \lor \formulaof{1}) \land (\formulaof{0} \lor \formulaof{2})\Big)
            \end{align*}
        \end{itemize}


    \end{frame}

    \begin{frame}{Clauses and conjunctive normal form}

        \begin{itemize}
            \item Atomic formulas and their negations are called {\bf positive and negative literals}.
            For example:
            \begin{align*}
                \lnot\catvariableof{B} \quad \text{is a negative literal}
                \quad, \quad
                \catvariableof{J} \quad \text{is a positive literal}
            \end{align*}
            \item {\bf Clauses} are disjunctions of literals.
            For example:
            \begin{align*}
                \lnot\catvariableof{B} \lor \catvariableof{J} \lor \catvariableof{M}
            \end{align*}
            \item A syntactical representation of a formula is in the {\bf conjunctive normal form}, if it is a conjunction of clauses (a knowledge base of clauses).
        \end{itemize}
    \end{frame}

    \begin{frame}{Parametrization of clauses}

        We parametrize a clause by
        \begin{itemize}
            \item a set $\hardlegset$ marking the atomic formulas appearing in the clause
            \item indices $\headindexof{\hardlegset}\in[2]^{\cardof{\hardlegset}}$ storing whether the literal is positive or negative
            \begin{align*}
                \headindexof{\catenumerator}
                = \begin{cases}
                      0 & \text{If the literal is positive, i.e. $\catvariableof{\catenumerator}$ appearing in the clause} \\
                      1 & \text{If the literal is negative, i.e. $\lnot\catvariableof{\catenumerator}$ appearing in the clause}
                \end{cases}
            \end{align*}
        \end{itemize}
        and denote it by $\clauseof{\hardlegset,\headindexof{\hardlegset}}$.

        \vfill

        The tenosr to the clause is decomposed as:
        \begin{align*}
            \clauseof{\hardlegset,\headindexof{\hardlegset}}
            = \underbrace{\onesat{\catvariableof{\hardlegset}}}_{\text{always satisfied}}
            - \underbrace{\bigotimes_{\catenumerator\in\hardlegset} \onehotmapofat{\headindexof{\catenumerator}}{\catvariableof{\catenumerator}}}_{\text{except for the world $\headindexof{\hardlegset}$}}
        \end{align*}
        \begin{center}
            {\bf We exploit this structure for efficient storage and processing of clauses.}
        \end{center}

    \end{frame}

    \begin{frame}{Conjunctive normal form (CNF)}
        Using the rules of replacement, any syntactical representation of a formula (involving only unary and binary connectives) can be transfered into the CNF.
        \medskip \\
        {\bf Example:} The formula $\formula = \left(\catvariableof{B} \lor \catvariableof{E}\right) \Rightarrow \left(\catvariableof{J} \lor \catvariableof{M}\right)$
        has a CNF with the clauses $\left(\lnot\catvariableof{B} \lor \catvariableof{B} \lor \catvariableof{E}\right)$ and $\left(\lnot\catvariableof{E} \lor \catvariableof{B} \lor \catvariableof{E}\right)$.
        We find the CNF by the rules of equivalence:
        \begin{align*}
            &\left(\catvariableof{B} \lor \catvariableof{E}\right) \Rightarrow \left(\catvariableof{J} \lor \catvariableof{M}\right)  & \text{(eliminate the implication)}\\
            &\equiv \lnot \left(\catvariableof{B} \lor \catvariableof{E}\right) \lor \left(\catvariableof{J} \lor \catvariableof{M}\right) & \text{(apply De Morgan)}\\
            &\equiv \left(\lnot\catvariableof{B} \land \lnot\catvariableof{E}\right) \lor \left(\catvariableof{J} \lor \catvariableof{M}\right) & \text{(apply the distributive law)}\\ % distributive
            &\equiv \underbrace{\left(\lnot\catvariableof{B} \lor \catvariableof{J} \lor \catvariableof{M}\right)}_{\text{first clause}}
            \land \underbrace{\left(\lnot\catvariableof{E} \lor \catvariableof{J} \lor \catvariableof{M}\right)}_{\text{second clause}}&
        \end{align*}

    \end{frame}

    \begin{frame}[fragile]{Representation of Syntax}

        In \tnreason, propositional syntax is represented by \emph{nested lists of strings}, for example:
        \begin{minted}{python}
    ["and", ["not", "Rained"], ["imp", "Rained", "Wet"]]
        \end{minted}
        \medskip

        Connectives $\exconnective$ get assigned a string $\scriptrepof{\exconnective}$:
        \begin{center}

                    \begin{tabular}{|c|c|}
                        \hline
                        \textbf{Unary connective $\exconnective$} & \textbf{$\scriptrepof{\exconnective}$} \\
                        \hline
                        $\lnot$                                   & \stringof{not}                         \\
                        \hline
                        $()$                                      & \stringof{id}                          \\
                        \hline
                    \end{tabular}};
                \node [anchor=center] at (5,0) {
                    \begin{tabular}{|c|c|}
                        \hline
                        \textbf{Binary connective $\exconnective$} & \textbf{$\scriptrepof{\exconnective}$} \\
                        \hline
                        $\land$                                    & \stringof{and}                         \\
                        \hline
                        $\lor$                                     & \stringof{or}                          \\
                        \hline
                        $\Rightarrow$                              & \stringof{imp}                         \\
                        \hline
                        $\oplus$                                   & \stringof{xor}                         \\
                        \hline
                        $\Leftrightarrow$                          & \stringof{eq}                          \\
                        \hline
                    \end{tabular}};
            \end{tikzpicture}
        \end{center}

        Any other string is interpreted by an atomic formula (a so-called propositional symbol).

    \end{frame}

    \begin{frame}{Propositional syntax in \tnreason{}}
        Having strings representing
        \begin{itemize}
            \item Connectives (by defined representations)
            \item Atomic Formulas (all other strings)
        \end{itemize}
        we represent formulas $\formulaof{1}\exconnective\formulaof{2}$
        \begin{centeredscript}
            $\scriptrepof{\formulaof{1}\exconnective\formulaof{2}}$ = [$\scriptrepof{\exconnective}$, $\scriptrepof{\formulaof{1}}$, $\scriptrepof{\formulaof{2}}$]
        \end{centeredscript}
        where we apply the conventions
        \begin{itemize}
            \item Connectives are at the 0th position in each list
            \item Further entries are either atoms as strings or encoded formulas itself
        \end{itemize}
    \end{frame}

    \begin{frame}{Examples of representations}
        Atomic variable $\var{Rained}$ by
        \begin{centeredscript}
            $\scriptrepof{\var{Rained}}$ = \pyinline{"Rained"}
        \end{centeredscript}
        Negative literal $\lnot\var{Rained}$ by
        \begin{centeredscript}
            $\scriptrepof{\lnot\var{Rained}}$ = \pyinline{["not","Rained"]}
        \end{centeredscript}
        Horn clause $\left(\var{Rained}\Rightarrow\var{Wet}\right)$ by
        \begin{centeredscript}
            $\scriptrepof{\var{Rained}\Rightarrow\var{Wet}}$ = \pyinline{["imp","Rained","Wet"]}
        \end{centeredscript}
        Knowledge Base
        $\kb=(\lnot\var{Rained})\land(\var{Rained}\Rightarrow\var{Wet})$ by
        \begin{centeredscript}
            $\scriptrepof{\kb}$ = \pyinline{["and", ["not", "Rained"], ["imp", "Rained", "Wet"]]}
        \end{centeredscript}
    \end{frame}


    \begin{frame}[fragile]{Basis encoding decompositions}
        The recursive structure of the nested lists \pyinline{expression}=$\scriptrepof{\exformula}$ is exploited in finding tensor network representations of $\exformula$.
        \begin{minted}{python}
    from tnreason.representation import formula_to_cores as ftc
    coreDict = ftc.create_formula_computation_cores(expression)
        \end{minted}
        The function creates a tensor networks of basis encodings to connectives by:
        \begin{itemize}
            \item If $\scriptrepof{\exformula}$ a list, add $\bencodingof{\scriptrepof{\exformula}[0]}$ and add tensors from recursive calls to the subformulas $\scriptrepof{\exformula}[1:]$
            \item Return empty dictionary, when $\scriptrepof{\exformula}$ is a string (then an atom)
        \end{itemize}
    \end{frame}
