    \begin{frame}{How to represent formulas as tensor networks?}

        \begin{block}{Multilinearity of tensor networks}
            Tensor networks are multilinear, that is linear in each core tensor decorating an edge.
        \end{block}

        The logical conjunction is the only connective function which is multilinear.

        \begin{block}{Non-linearity problem}
            How to represent non-linear formulas like $\lnot\catvariable$ in a tensor network?
        \end{block}

            {\bf Solution:} \\
        We develop the \emph{basis calculus} scheme instead of \emph{coordinate calculus}.
        % \end{center}

    \end{frame}

    \begin{frame}{Coordinate and basis encoding}

        {\bf Coordinate encoding:} \\
        Store an information $\formulaev{\shortcatindices}$ in the coordinates of a tensor
        \begin{align*}
            \formulaat{\shortcatvariables}
            = \sum_{\shortcatindices\in[2]^{\catorder}}\formulaev{\shortcatindices} \cdot \onehotmapofat{\shortcatindices}{\shortcatvariables}
        \end{align*}

            {\bf Basis encoding:} \\
        Store an information $\formulaev{\shortcatindices}$ in a basis vector $\onehotmapofat{\formulaev{\shortcatindices}}{\headvariableof{\formula}}$ to an auxiliary variable $\headvariableof{\formula}$, that is
        \begin{align*}
            \bencodingofat{\formula}{\headvariableof{\formula},\shortcatvariables}
            = \sum_{\shortcatindices\in[2]^{\catorder}}  \onehotmapofat{\formulaev{\shortcatindices}}{\shortcatvariables} \otimes \onehotmapofat{\shortcatindices}{\shortcatvariables}
        \end{align*}

    \end{frame}

    \begin{frame}{Basis calculus}
        Idea of basis calculus:
        \begin{itemize}
            \item Represent input $\shortcatindices$ by one-hot encoding $\onehotmapofat{\shortcatindices}{\shortcatvariables}$
            \item Contract with the basis encoding of a function
            \item Get one-hot encoding of $\formula(\shortcatindices)$ as a result
        \end{itemize}

        \begin{block}{Basis calculus scheme}
            States are represented by basis vectors (their one-hot encodings) and processed by contractions with basis encodings.
        \end{block}
    \end{frame}

    \begin{frame}{\DecompositionSparsity{}}

        \begin{theorem}
            The basis encoding of a composed formula is the contraction of the basis encodings to its connectives.
        \end{theorem}
        \textbf{Proof sketch:} When contracting the decomposition with one-hot encodings of any $\shortcatindices\in[2]^{\catorder}$, we get by basis calculus the one-hot encodings of the composition formulas, and finally the one-hot encoding of the whole formula.
        This coincides with contracting the full basis encoding since
        \begin{align*}
            \contractionof{\bencodingofat{\formula}{\headvariableof{\exformula},\shortcatvariables},\onehotmapofat{\shortcatindices}{\shortcatvariables}}{
                \headvariableof{\exformula}
            }
            = \onehotmapofat{\formulaat{\indexedshortcatvariables}}{\headvariableof{\formula}} \, .
        \end{align*}

        \begin{block}{\DecompositionSparsity{}}
            When a function can be decomposed into connective functions, its corresponding tensor has a tensor network decomposition by the basis encodings of the connective functions.
%            When there is a syntactical representation of a formula, we find a tensor network decomposition of the formula.
        \end{block}
    \end{frame}

    \begin{frame}{Formula evaluation by message passing of one-hot encodings}

        \only<1-5>{
            \emph{Basis Calculus:} Computation by basis vectors
            \begin{itemize}
                \item Input states are prepared by their one-hot encodings
                \item Contractions with the basis encodings result in the one-hot encoding of the output
            \end{itemize}
                {\bf Advantage:} Decomposition of basis encodings enables to decompose global contractions into local contractions.
        }
        \par

        \only<6>{
            This is a first example of a \emph{message passing scheme:}
            \begin{itemize}
                \item {\bf Messages} are computed by contractions of sub-networks (two blue contractions, one red contraction)
                \item {\bf Passing} of previous messages (blue contractions) to compute new messages (red contraction)
            \end{itemize}
        }
    \end{frame}

    \begin{frame}[fragile]{Representation of Syntax}

        In \tnreason, propositional syntax is represented by \emph{nested lists of strings}, for example:
        \begin{minted}{python}
    ["and", ["not", "Rained"], ["imp", "Rained", "Wet"]]
        \end{minted}
        \medskip

        Connectives $\exconnective$ get assigned a string $\scriptrepof{\exconnective}$:
        \begin{center}
            \begin{tikzpicture}
                \node [anchor=center] at (0,0) {
                    \begin{tabular}{|c|c|}
                        \hline
                        \textbf{Unary connective $\exconnective$} & \textbf{$\scriptrepof{\exconnective}$} \\
                        \hline
                        $\lnot$                                   & \stringof{not}                         \\
                        \hline
                        $()$                                      & \stringof{id}                          \\
                        \hline
                    \end{tabular}};
                \node [anchor=center] at (5,0) {
                    \begin{tabular}{|c|c|}
                        \hline
                        \textbf{Binary connective $\exconnective$} & \textbf{$\scriptrepof{\exconnective}$} \\
                        \hline
                        $\land$                                    & \stringof{and}                         \\
                        \hline
                        $\lor$                                     & \stringof{or}                          \\
                        \hline
                        $\Rightarrow$                              & \stringof{imp}                         \\
                        \hline
                        $\oplus$                                   & \stringof{xor}                         \\
                        \hline
                        $\Leftrightarrow$                          & \stringof{eq}                          \\
                        \hline
                    \end{tabular}};
            \end{tikzpicture}
        \end{center}

        Any other string is interpreted by an atomic formula (a so-called propositional symbol).

    \end{frame}

    \begin{frame}{Propositional syntax in \tnreason{}}
        Having strings representing
        \begin{itemize}
            \item Connectives (by defined representations)
            \item Atomic Formulas (all other strings)
        \end{itemize}
        we represent formulas $\formulaof{1}\exconnective\formulaof{2}$
        \begin{centeredscript}
            $\scriptrepof{\formulaof{1}\exconnective\formulaof{2}}$ = [$\scriptrepof{\exconnective}$, $\scriptrepof{\formulaof{1}}$, $\scriptrepof{\formulaof{2}}$]
        \end{centeredscript}
        where we apply the conventions
        \begin{itemize}
            \item Connectives are at the 0th position in each list
            \item Further entries are either atoms as strings or encoded formulas itself
        \end{itemize}
    \end{frame}

    \begin{frame}{Examples of representations}
        Atomic variable $\var{Rained}$ by
        \begin{centeredscript}
            $\scriptrepof{\var{Rained}}$ = \pyinline{"Rained"}
        \end{centeredscript}
        Negative literal $\lnot\var{Rained}$ by
        \begin{centeredscript}
            $\scriptrepof{\lnot\var{Rained}}$ = \pyinline{["not","Rained"]}
        \end{centeredscript}
        Horn clause $\left(\var{Rained}\Rightarrow\var{Wet}\right)$ by
        \begin{centeredscript}
            $\scriptrepof{\var{Rained}\Rightarrow\var{Wet}}$ = \pyinline{["imp","Rained","Wet"]}
        \end{centeredscript}
        Knowledge Base
        $\kb=(\lnot\var{Rained})\land(\var{Rained}\Rightarrow\var{Wet})$ by
        \begin{centeredscript}
            $\scriptrepof{\kb}$ = \pyinline{["and", ["not", "Rained"], ["imp", "Rained", "Wet"]]}
        \end{centeredscript}
    \end{frame}


    \begin{frame}[fragile]{Basis encoding decompositions}
        The recursive structure of the nested lists \pyinline{expression}=$\scriptrepof{\exformula}$ is exploited in finding tensor network representations of $\exformula$.
        \begin{minted}{python}
    from tnreason.representation import formula_to_cores as ftc
    coreDict = ftc.create_formula_computation_cores(expression)
        \end{minted}
        The function creates a tensor networks of basis encodings to connectives by:
        \begin{itemize}
            \item If $\scriptrepof{\exformula}$ a list, add $\bencodingof{\scriptrepof{\exformula}[0]}$ and add tensors from recursive calls to the subformulas $\scriptrepof{\exformula}[1:]$
            \item Return empty dictionary, when $\scriptrepof{\exformula}$ is a string (then an atom)
        \end{itemize}
    \end{frame}
