    \begin{frame}{Monotonicity of Entailment}

        \begin{theorem}[Monotonicity of Entailment]
            Given a knowledge base $\kb$ of formulas, let $\seckb\subset\kb$ be a subset and let $\formula$ be another propositional formula.
            If $\seckb\models\formula$ then also $\kb\models\formula$.
        \end{theorem}

            {\bf Proof sketch:} If $\seckb\models\formula$ then
        \begin{align*}
            \contractionof{\seckb[\shortcatvariables],\lnot\formulawith}{\shortcatvariables} = \zerosat{\shortcatvariables}
        \end{align*}
        and since this is a sub-contraction of $\contraction{\kbwith,\formulawith}$ also
        \begin{align*}
            \contractionof{\kbwith,\lnot\formulawith}{\shortcatvariables} = \zerosat{\shortcatvariables}
        \end{align*}
        and therefore $\kb\models\formula$.

        Notice that the back direction does not hold:
        When $\kb\models\formula$, then $\seckb\models\formula$ does not always hold.
    \end{frame}




    \begin{frame}{Invariance of adding entailled formulas}


        If $\kb\models\formula$, then for any $\shortcatindices\in[2]^\seldim$
        \begin{align*}
            &\kbat{\indexedshortcatvariables}
            = \begin{cases}
                  1 & \text{if $\shortcatindices$ is a model of $\kb$} \\
                  0 & \text{else}
            \end{cases} \\
            &\quad\quad = \begin{cases}
                              \kbat{\indexedshortcatvariables} \cdot \formulaat{\indexedshortcatvariables} &
                              \text{(since $\kb\models\formula$)} \\
                              \kbat{\indexedshortcatvariables} \cdot \formulaat{\indexedshortcatvariables} &
                              \text{(since $\kbat{\indexedshortcatvariables}=0$)}
            \end{cases} \, .
        \end{align*}
            {\bf Consequence:} When $\kb\models\formula$, then
        \begin{align*}
            \kbwith
            = \contractionof{\kbwith,\formulawith}{\shortcatvariables} \, .
        \end{align*}

        \begin{block}{Interpretation: Memory of entailed formulas}
            We can add entailed formulas to a knowledge base.
        \end{block}

    \end{frame}


    \begin{frame}{Usage for efficient reasoning}
        \par
        {\bf Efficiency:} If $\seckb$ is a small subset, $\seckb\models\formula$ can be decided efficiently.

            {\bf Reuse:} Entailed formulas $\formula$ can be added to the knowledge base and used in further $\seckb$ choices.

        \begin{itemize}
            \item \emph{Messages:} Entailed formulas $\formula$ "summarize" subsets and are added to knowledge bases
            \item \emph{Propagation:} Entailed formulas are propagated throughout the knowledge base and
        \end{itemize}


        \begin{block}{Design of inference algorithms}
            Logical inference algorithms are designed based on the monotonicity and memory properties:
            \begin{itemize}
                \item Choice of pairs $\seckb,\formula$ such that $\seckb\models\formula$: Often syntactical rules are applied
                \item Priority of adding $\formula$ to $\kb$: Need to apply restrictions to avoid overloads
            \end{itemize}
        \end{block}


    \end{frame}

    \begin{frame}{Modus Ponens}


        \begin{block}{Modus Ponens Inference Rule}
            {\bf Modus Ponens} is the simplest inference rule: When an implication holds and the premise is true, then the head follows.
            \begin{align*}
                \frac{
                    (\catvariableof{0}\land\ldots\land\catvariableof{\catorder-1})\rightarrow\catvariableof{\catorder} \,,\,\, (\catvariableof{0}\land\ldots\land\catvariableof{\catorder-1})
                }{
                    \catvariableof{\catorder}
                }
            \end{align*}
        \end{block}

    \begin{frame}{Clauses}
        \par
        {\bf Recall:} Given $\hardlegset\subset[\catorder]$ and $\headindexof{\hardlegset}\in\bigtimes_{\catenumerator\in\hardlegset}[2]$ the corresponding clause is
        \begin{align*}
            \clauseofat{\hardparam}{\catvariableof{\hardlegset}}
            = \onesat{\catvariableof{\hardlegset}} - \onehotmapofat{\headindexof{\hardlegset}}{\catvariableof{\hardlegset}}
        \end{align*}

        Special types:
        \begin{itemize}
            \item \emph{Horn clause:} There is at most one $\catenumerator\in\hardlegset$ with $\headindexof{\catenumerator}=1$
            \item \emph{Definite clause:} There is at exactly one $\catenumerator\in\hardlegset$ with $\headindexof{\catenumerator}=1$
        \end{itemize}

        Interpretation of definite clauses: Implications \\

        {\bf Example:} If in addition to $F$ being present on an invoice, an invoice came from buisness partner $BP1$, then its booked using account $A1$.
        \begin{align*}
            \Big(\catvariableof{F} \land \catvariableof{BP1} \rightarrow \catvariableof{A1}\Big)
            &\Leftrightarrow
            \lnot \Big((\catvariableof{F} \land \catvariableof{BP1}) \lor \catvariableof{A1} \Big) \\
            &\Leftrightarrow
            \Big(\lnot\catvariableof{F} \lor \lnot \catvariableof{BP1} \lor \catvariableof{A1} \Big)
        \end{align*}

    \end{frame}

    \begin{frame}{Iteration of modus ponens inference rule}
        \par
        {\bf Given:} A knowledge base of implications (definite clauses) in the form
        \begin{align*}
            \Big(\bigwedge_{\catenumerator\in\mathcal{M}}\catvariableof{\catenumerator}\Big)\rightarrow\catvariableof{\seccatenumerator}
        \end{align*}
        which are clauses
        \begin{align*}
            \Big(\bigvee_{\catenumerator\in\mathcal{M}}\lnot\catvariableof{\catenumerator}\Big) \lor \catvariableof{\seccatenumerator}
            \quad \text{parametrized by} \quad \hardparam =\big(\mathcal{M}\cup\{\seccatenumerator\},(\onetuple{\mathcal{M}},0)\big)
            \, .
        \end{align*}
        \medskip
        We iterate:
        \begin{itemize}
            \item Computation of {\bf messages}: The formula $\catvariableof{\catorder}$ is entailed by the local knowledge base
            \begin{align*}
                \seckb = \Big\{\Big(\bigwedge_{\catenumerator\in\mathcal{M}}\catvariableof{\catenumerator}\Big)\rightarrow\catvariableof{\seccatenumerator}\Big\}
                \cup \Big\{\catvariableof{\catenumerator} \wcols \catenumerator\in\mathcal{M} \}
            \end{align*}
            \item The message $\catvariableof{\seccatenumerator}$ is {\bf propagated} and used for further local inferences.
        \end{itemize}

    \end{frame}

    \begin{frame}[fragile]{Unit propagation Algorithm}
        \begin{algorithm}[H]
            \begin{algorithmic}
                \Require Set of clauses $\mathcal{C}$ (each parameterized by $\hardparam$)
                \Ensure Entailed literals $\mathcal{L}$, remaining set of clauses
                \iosepline
                \State Initialize $\mathcal{L}=\varnothing$
                \While{exists a clause $(\{\node\},\secheadindexof{\node})$ of length 1 in $\mathcal{C}$}
                    \For{all clauses $\hardparam$ with $\node\in\hardlegset$}
                        \If{$\headindexof{\node}\neq\secheadindexof{\node}$}
                            \hfill \textcolor{blue}{(then the clause shrinks)}
                            %\hfill \textcolor{blue}{(then the literal is satisfied)}
                            \State $\hardparam\algdefsymbol(\hardlegset\backslash\{\node\},\headindexof{\hardlegset\backslash\{\node\}})$
                        \Else
                            \hfill \textcolor{blue}{(then the clause becomes trivial)}
                            \State $\mathcal{C}.\mathrm{drop(}\hardparam\mathrm{)}$ %\hfill \textcolor{blue}{(then the contracted clause is trivial)}
                        \EndIf
                    \EndFor
                    \State $\mathcal{L}.\mathrm{append(}\{\node\},\secheadindexof{\node}\mathrm{)}$ \hfill \textcolor{blue}{(the literal is entailed and added to solution)}
                \EndWhile
            \end{algorithmic}
        \end{algorithm}
    \end{frame}

        Unit propagation is {\bf sound}, that is for any clause $\big(\{\node\},\headindexof{\node}\big)\in\mathcal{L}$ in the solution we have

    \begin{frame}{Definite Clauses}

        To show completeness, we need the following property of definite clauses:

        \begin{lemma}
            A knowledge base of definite clauses of length at least $2$ always has the states $\onetuple{\nodes}$ and $\zerotuple{\nodes}$ as models.
        \end{lemma}
            {\bf Proof sketch:} By using the contraction of clauses properties we know that:
        \begin{itemize}
            \item When contracting with $\bigtimes_{\nodein}\tbasisat{\catvariableof{\node}}$, all clauses become trivial.
            \item When contracting with $\bigtimes_{\nodein}\fbasisat{\catvariableof{\node}}$, all clauses become trivial.
        \end{itemize}
        Thus, $\catindexof{\nodes}=\onetuple{\nodes}$ and $\catindexof{\nodes}=\zerotuple{\nodes}$ are models of the knowledge base.


%        Each knowledge base of definite clause has $\catindexof{\nodes}=\onetuple{\nodes}$ as a model.

%        If in addition the length of each clause is at least 2, then also $\zerotuple{\nodes}$ as a model.

    \end{frame}


    \begin{frame}{Unit propagation is complete on definite clauses}
        \begin{theorem}
            When $\kb$ is a knowledge base of definite clauses, then $\mathcal{L}$ contains all entailed literals.
        \end{theorem}
        \medskip

        \textbf{Sketch of the proof:}
        \begin{itemize}
            \item Unit propagation on definite clauses terminates with a knowledge base of definite clauses of length at least $2$
            \item The final knowledge base has $\onetuple{\nodes}$ and $\zerotuple{\nodes}$ as a model
            \item The final knowledge base does therefore not entail for any $\node\in\nodes$ the literals $\catvariableof{\node}$ or $\lnot\catvariableof{\node}$
        \end{itemize}
    \end{frame}

    \begin{frame}{Sub-contractions}

            {\bf Problem:} When keeping not all variables open, contractions may fail to be propositional formulas.
    For example:
        \begin{align*}
            \contractionof{\catvariableof{0}\Rightarrow\catvariableof{1}}{\catvariableof{0}}
            = \contractionof{\coloredmatrixof{1 & 1 \\ 0 & 1}{\catvariableof{0},\catvariableof{1}}}{\catvariableof{0}}
            = \coloredmatrixof{2 \\ 1 }{\catvariableof{0}}
        \end{align*}
\medskip
            {\bf Solution:} Build the support of the contracted tensor.
        We define the \emph{support $\nonzeroof{\hypercorewith}$} of a tensor $\hypercorewith$, as a tensor with coordinates
        \begin{align*}
            \nonzeroof{\hypercoreat{\indexedshortcatvariables}}
            = \begin{cases}
                  1 & \ifspace \hypercoreat{\indexedshortcatvariables}\neq 0 \\
                  0 & \ifspace \hypercoreat{\indexedshortcatvariables} = 0
            \end{cases} \, .
        \end{align*}
        The support therefore marks the non-vanishing coordinates by a $1$.
    \end{frame}


    \begin{frame}{Support of sub-contractions}

        The support of the sub-contraction $\seckb\subset\kb$ leaving $\secnodes\subset\nodes$ open is a formula
        \begin{align*}
            \formulaofat{(\seckb,\secnodes)}{\catvariableof{\secnodes}}
            := \nonzeroof{\contractionof{\seckb}{\catvariableof{\secnodes}}} \, .
        \end{align*}

        The support of a sub-contraction is the \emph{minimal} formula $\formula$ containing the atomic variables $\catvariableof{\secnodes}$ such that $\seckb \models \formula$.
        That is:
        \begin{align*}
            \formulaofat{(\seckb,\secnodes)}{\catvariableof{\secnodes}}
            = \argmin_{\formulaof{\catvariableof{\secnodes}} \wcols \seckb \models \formula} \contraction{\formulaof{\catvariableof{\secnodes}}} \, .
        \end{align*}

        \begin{block}{Interpretation}
            The support of a sub-contraction is the \emph{best summary} of the knowledge about $\seckb$ using only the variables $\catvariableof{\secnodes}$.
        \end{block}

    \end{frame}

            For any propositional formulas $\formulaof{0},\formulaof{1}$ and an atomic variable $\catvariable$ the resolution inference rule is sound:
            \begin{align*}
                \frac{
                    \formulaof{0} \lor \catvariable \,,\,\, \formulaof{1} \lor \lnot\catvariable
                }{
                    \formulaof{0} \lor \formulaof{1}
                }
            \end{align*}
        \end{block}

    \begin{frame}{Completeness of resolution}
        \par
        {\bf Deciding satisfiability:}
        \begin{itemize}
            \item Entailment $\kb\models\formula$ is equal to the unsatisfiability of $\kb\land\lnot\formula$.
            \item Can we detect an unsatisfiable ("inconsistent") knowledge base?
        \end{itemize}

        Detecting unsatisfiability:
        \begin{itemize}
            \item Suppose $\catvariableof{\node}$ and $\lnot\catvariableof{\node}$ are in the knowledge base (e.g. added by resolution)
            \item Understand them as $\zerosat{\catvariableof{\node}}\lor\catvariableof{\node}, \, \zerosat{\catvariableof{\node}}\lor\lnot\catvariableof{\node},$
            \item Resolution then includes an unsatisfiable $\zerosat{\catvariableof{\node}}$ and detects therefore the inconsistency.
        \end{itemize}

        \begin{theorem}
            When a knowledge base in $CNF$ is unsatisfiable, then the unsatisfiability is detected by iterative resolution.
        \end{theorem}
    \end{frame}

    \begin{frame}{Scheduling inference steps}

        \begin{block}{Number of large contractions}
            The number of possible sub-contractions grows fast with the number $k$ of the subcontraction: $\binom{\cardof{\edges}}{k}$
        \end{block}
        \begin{center}
            \bf The choice and scheduling of inference steps is of central importance for efficient algorithms!
        \end{center}

        Resolution:
        \begin{itemize}
            \item Restricts the possible sub-contractions to those of clauses with complementary clauses
            \item Requires further scheduling policies for efficiency
        \end{itemize}

        Selection of contractions to be performed is central to algorithm design
        \begin{itemize}
            \item Forward chaining: Unit propagation with positive literals only
            \item SLD resolution (e.g. in ProLog)
        \end{itemize}

    \end{frame}
