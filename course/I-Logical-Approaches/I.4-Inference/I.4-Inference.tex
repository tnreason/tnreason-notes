    \section{Monotonicity of Entailment}

    \begin{frame}{Motivation}
        Contractions of large knowledge bases get infeasible.
        \begin{center}
            \bf Is it really necessary to perform large contractions?
        \end{center}

        \begin{block}{Example: Entailment in the toy accounting system}
            When knowing that feature $\catvariableof{F}$ is true, we can conclude that $\catvariableof{A1}$ is true, without considering other accounting rules.
        \end{block}

    \end{frame}


    \begin{frame}{Outlook: Inference algorithms}
        Inference algorithms focus on subsets of knowledge bases.

        We now show that:
        \begin{itemize}
            \item {\bf Soundness:} "Every by a sub knowledge base entailed formula is entailed by the knowledge base itself."
            \item {\bf Memory:} "Every entailed formula can be added to the knowledge base."
        \end{itemize}

    \end{frame}

    \begin{frame}{Monotonicity of Entailment}

        \begin{theorem}[Monotonicity of Entailment]
            Given a knowledge base $\kb$ of formulas, let $\seckb\subset\kb$ be a subset and let $\formula$ be another propositional formula.
            If
            \begin{align*}
                \seckb\models\formula
            \end{align*}
            then also
            \begin{align*}
                \kb\models\formula
            \end{align*}
        \end{theorem}

    \end{frame}


    \begin{frame}{Soundness but no Completeness}
        Notice that the back direction does not hold:
        When $\kb\models\formula$, then $\seckb\models\formula$ does not always hold.


    \end{frame}

    \begin{frame}{Invariance of adding entailled formulas}

        When $\kb\models\formula$, then
        \begin{align*}
            \kbwith
            = \contractionof{\kbwith,\formulawith}{\shortcatvariables}
        \end{align*}

        \begin{block}{Interpretation}
            We can add memory formulas to a knowledge base.
        \end{block}

    \end{frame}

    \begin{frame}{Support tensors}

        Support tensor to a tensor $\hypercorewith$: $\nonzeroof{\hypercorewith}$ with coordinates
        \begin{align*}
            \nonzeroof{\hypercoreat{\indexedshortcatvariables}}
            = \begin{cases}
                  1 & \ifspace \hypercoreat{\indexedcatvariables}\neq 0 \\
                  0 & \text{else}
            \end{cases}
        \end{align*}

        \begin{block}{Invariance under adding support tensor to sub-contractions}
            Any support tensor of a sub-contraction can be added to a tensor network: When $\secgraph$ a sub-hypergraph of $\graph$, then
            \begin{align*}
                \contractionof{\extnetwith}{\nodevariables}
                = \contractionof{\extnetwith,
                    \nonzeroof{\contractionof{\tnetofat{\secgraph}{\nodevariables}}{\catvariableof{\secnodes}}}
                }{\nodevariables}
            \end{align*}
        \end{block}

        In propositional logics, this means:
        \begin{align*}
            \nonzeroof{\contractionof{\extnetwith}{\nodevariables}}
            \models \nonzeroof{\contractionof{\tnetofat{\secgraph}{\nodevariables}}{\catvariableof{\secnodes}}}
        \end{align*}


    \end{frame}


    \begin{frame}{Inference algorithms based on sub-contractions}


        Computation by sub-contractions (including previous messages)

        \begin{block}{Problem: Many possible sub-contractions}
            In large tensor networks, there are many possible sub-contractions.
            For efficiency, different scheduling policies emerge:
            \begin{itemize}
                \item Syntactical inference algorithms (today)
                \item Message passing algorithms (next lecture)
            \end{itemize}
        \end{block}

    \end{frame}


    \section{Syntactical Inference Rules}

    \begin{frame}{Inference using Syntax}

        So far: Inference and Entailment is formulated by semantics (with tensor interpretations).

        Syntactical Inference Rules find entailed formulas using syntax only.

        We here derive them as instances of sub-contractions support adding.

        Notation: When $\formula$ is entailed by $\seckb$ we denote
        \begin{align*}
            \frac{
                \seckb
            }{
                \exformula
            }
        \end{align*}

    \end{frame}

    \begin{frame}{Clauses}

        Given $\hardlegset\subset[\catorder]$ and $\headindexof{\hardlegset}\in\bigtimes_{\catenumerator\in\hardlegset}[2]$ the corresponding clause is
        \begin{align*}
            \clauseofat{\hardparam}{\catvariableof{\hardlegset}}
            = \onesat{\catvariableof{\hardlegset}} - \onehotmapofat{\headindexof{\hardlegset}}{\catvariableof{\hardlegset}}
        \end{align*}

        Special types:
        \begin{itemize}
            \item Horn clause: There is at most one $\catenumerator\in\hardlegset$ with $\headindexof{\catenumerator}=1$
            \item Definite clause: There is at exactly one $\catenumerator\in\hardlegset$ with $\headindexof{\catenumerator}=1$
        \end{itemize}

        Interpretation of definite clauses: Implications
        Example:
        \begin{align*}
            \Big(\catvariableof{F} \land \catvariableof{GP1} \rightarrow \catvariableof{MAT}\Big)
            &\Leftrightarrow
            \lnot \Big((\catvariableof{F} \land \catvariableof{GP1}) \lor \catvariableof{MAT} \Big) \\
            &\Leftrightarrow
            \Big(\lnot\catvariableof{F} \lor \lnot \catvariableof{GP1} \lor \catvariableof{MAT} \Big)
        \end{align*}

    \end{frame}



    \begin{frame}{Modus Ponens}

        {\bf Modus Ponens} is the simplest inference rule: When an implication holds and the premise is true, then the head follows.
        When $\catvariableof{0}\rightarrow\catvariableof{1}$ and $\catvariableof{0}$ is true then also $\catvariableof{1}$.

        \begin{block}{Modus Ponens Inference Rule}

            \begin{align*}
                \frac{
                    (\catvariableof{0}\land\ldots\land\catvariableof{\catorder-1})\rightarrow\catvariableof{\catorder} \,,\,\, (\catvariableof{0}\land\ldots\land\catvariableof{\catorder-1})
                }{
                    \catvariableof{\catorder}
                }
            \end{align*}
        \end{block}

    \section{Forward Chaining in Horn Logic}




    \begin{frame}[fragile]{Forward Chaining}
        \begin{algorithm}[H] % Force 'Here' to disable floating
            \begin{algorithmic}
                \Require Set of clauses $\mathcal{C}$
                \Ensure Entailed literals $\mathcal{L}$, remaining set of clauses
                \iosepline
                \State Initialize $\mathcal{L}=\varnothing$
                \While{exists a clause $(\{\node,\secheadindexof{\node}\})$ of length 1 in $\mathcal{C}$}
                    \For{all clauses $\hardparam$ with $\node\in\hardlegset$}
                        \If{$\headindexof{\node}\neq\secheadindexof{\node}$}
                            \State $\hardparam\algdefsymbol(\hardlegset\backslash\{\node\},\headindexof{\hardlegset\backslash\{\node\}})$
                        \Else
                            \State $\mathcal{C}.\mathrm{drop(}\hardparam\mathrm{)}$
                        \EndIf
                    \EndFor
                    \State $\mathcal{L}.\mathrm{append(}\{\node,\secheadindexof{\node}\}\mathrm{)}$
                \EndWhile
            \end{algorithmic}
        \end{algorithm}
    \end{frame}

    \begin{frame}{Definite Clauses}
        Each knowledge base of definite clause has $\catindexof{\nodes}=\onetuple{\nodes}$ as a model.

        If in addition the length of each clause is at least 2, then also $\zerotuple{\nodes}$ as a model.

    \end{frame}


    \begin{frame}{Forward chaining is complete on definite clauses}
        \begin{theorem}
            When $\kb$ is all clauses in are definite, i.e. have exactly one positive literal, then $\mathcal{L}$ contains all entailed literals
        \end{theorem}
        \medskip

        \textbf{Sketch of the proof:}
        \begin{itemize}
            \item Forward chaining on definite clauses terminates with a knowledge base of definite clauses of length at least $2$
            \item The final knowledge base has $\onetuple{\nodes}$ and $\zerotuple{\nodes}$ as a model
            \item The final knowledge base does therefore not entail for any $\node\in\nodes$ the literals $\catvariableof{\node}$ or $\lnot\catvariableof{\node}$
        \end{itemize}
    \end{frame}

    \begin{frame}{Scheduling is key}

        \begin{block}{Number of large contractions}
            The number of possible sub-contractions grows fast with the number $k$ of the subcontraction: $\binom{\cardof{\edges}}{k}$
        \end{block}

        Resolution:
        \begin{itemize}
            \item Restricts the possible sub-contractions to those of clauses with complementary clauses
            \item Requires further scheduling policies for efficiency
        \end{itemize}

        Selection of contractions to be performed is central to algorithm design
        \begin{itemize}
            \item Forward chaining
            \item SLD resolution (e.g. in ProLog)
        \end{itemize}

    \end{frame}
