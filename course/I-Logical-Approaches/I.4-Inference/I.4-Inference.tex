    \section{Montonocity of Entailment}

    \begin{frame}{Motivation}
        Contractions of large knowledge bases get infeasible.

        Inference algorithms focus on subsets of knowledge bases.

        We now show that:
        \begin{itemize}
            \item {\bf Soundness:} "Every by a sub knowledge base entailed formula is entailed by the knowledge base itself."
            \item {\bf Memory:} "Every entailed formula can be added to the knowledge base."
        \end{itemize}

    \end{frame}

    \begin{frame}{Montonocity of Entailment}

        \begin{theorem}[Monotonicity of Entailment]
            Given a knowledge base $\kb$ of formulas, let $\seckb\subset\kb$ be a subset and let $\formula$ be another propositional formula.
            If
            \begin{align*}
                \seckb\models\formula
            \end{align*}
            then also
            \begin{align*}
                \kb\models\formula
            \end{align*}
        \end{theorem}

    \end{frame}


    \begin{frame}{Soundness but no Completeness}
        Notice that the back direction does not hold:
        When $\kb\models\formula$, then $\seckb\models\formula$ does not always hold.


    \end{frame}

    \begin{frame}{Invariance of adding entailled formulas}

        When $\kb\models\formula$, then
        \begin{align*}
            \kbwith
            = \contractionof{\kbwith,\formulawith}{\shortcatvariables}
        \end{align*}

        \begin{block}{}
        \end{block}

    \end{frame}

    \begin{frame}{Support tensors}

        Support tensor to a tensor $\hypercorewith$: $\nonzeroof{\hypercorewith}$ with coordinates
        \begin{align*}
            \nonzeroof{\hypercoreat{\indexedshortcatvariables}}
            = \begin{cases}
                  1 & \ifspace \hypercoreat{\indexedcatvariables}\neq 0 \\
                  0 & \text{else}
            \end{cases}
        \end{align*}

        \begin{block}{Invariance under adding support tensor to sub-contractions}
            Any support tensor of a sub-contraction can be added to a tensor network: When $\secgraph$ a sub-hypergraph of $\graph$, then
            \begin{align*}
                \contractionof{\extnetwith}{\nodevariables}
                = \contractionof{\extnetwith,
                    \nonzeroof{\contractionof{\tnetofat{\secgraph}{\nodevariables}}{\catvariableof{\secnodes}}}
                }{\nodevariables}
            \end{align*}
        \end{block}

        In propositional logics, this means:
        \begin{align*}
            \nonzeroof{\contractionof{\extnetwith}{\nodevariables}}
            \models \nonzeroof{\contractionof{\tnetofat{\secgraph}{\nodevariables}}{\catvariableof{\secnodes}}}
        \end{align*}


    \end{frame}

    \begin{frame}{Modus Ponens}
