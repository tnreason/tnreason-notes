    \addtask{1}{
        The formulas translate into the following clauses:
        \begin{itemize}
            \item $\formulaof{0} \equiv\Big(\lnot P \lor Q\Big)$
            \item $\formulaof{1} \equiv\Big(\lnot L \lor \lnot M \lor P\Big)$
            \item $\formulaof{2} \equiv\Big(\lnot B \lor \lnot L \lor M\Big)$
            \item $\formulaof{3} \equiv\Big(\lnot A \lor \lnot P \lor L\Big)$
            \item $\formulaof{4} \equiv\Big(\lnot A \lor \lnot B \lor L\Big)$
            \item $\formulaof{5} \equiv\big(A\big)$
            \item $\formulaof{6} \equiv\big(B\big)$
        \end{itemize}
        The $CNF$ of the knowledge base is given by the conjunction of these clauses.
    }
    \addtask{2}{
        Let us describe the iterations of the $\mathrm{while}$ loop in the unit propagation algorithm.
        The chosen unit clauses are always dropped from $\mathcal{C}$ and added to $\mathcal{L}$ (with the corresponding iteration number as index).
        \begin{itemize}
            \item \textbf{Iteration 1:} We choose the unit clause $\big(\{A\},0\big)$ (alternatively: $\big(\{B\},0\big)$).
            The clauses $\formulaof{3}$ and $\formulaof{4}$ are simplified by removing the literal $A$ and we get the new clauses
            \begin{align*}
                \formulaof{3} &\equiv\Big(\lnot P \lor L\Big) \, , \\
                \formulaof{4} &\equiv\Big(\lnot B \lor L\Big) \, .
            \end{align*}
            \item \textbf{Iteration 2:} We choose the unit clause $\big(\{B\},0\big)$ (alternatively: $\big(\{A\},0\big)$).
            The clauses $\formulaof{2}$ and $\formulaof{4}$ are simplified by removing the literal $B$ and we get the new clauses
            \begin{align*}
                \formulaof{2} &\equiv\Big(\lnot L \lor M\Big) \, , \\
                \formulaof{4} &\equiv\Big(L\Big) \, .
            \end{align*}
            \item \textbf{Iteration 3:} We choose the only remaining unit clause $\big(\{L\},0\big)$.
            The clause $\formulaof{3}$ is dropped (since $L$ also appearing as positive literal).
            The clauses $\formulaof{1}$ and $\formulaof{2}$ are simplified by removing the literal $L$ and we get the new clauses
            \begin{align*}
                \formulaof{1} &\equiv\Big(\lnot M \lor P\Big) \, , \\
                \formulaof{2} &\equiv\Big(M\Big) \, .
            \end{align*}
            \item \textbf{Iteration 4:} We choose the only remaining unit clause $\big(\{M\},0\big)$.
            The clause $\formulaof{1}$ is simplified by removing the literal $M$ and we get the new clause
            \begin{align*}
                \formulaof{1} &\equiv\Big(P\Big) \, .
            \end{align*}
            \item \textbf{Iteration 5:} We choose the only remaining unit clause $\big(\{P\},0\big)$.
            The clause $\formulaof{0}$ is simplified to
            \begin{align*}
                \formulaof{0} &\equiv\Big(Q\Big) \, .
            \end{align*}
            \item \textbf{Iteration 6:} We choose the only remaining unit clause $\big(\{Q\},0\big)$.
            No further clauses are remaining.
        \end{itemize}
        In total we get the solution
        \begin{align*}
            \mathcal{L} = \Big\{
            \big(\{A\},0\big),\big(\{B\},0\big),\big(\{L\},0\big),
            \big(\{M\},0\big),\big(\{P\},0\big),\big(\{Q\},0\big)
            \Big\} \, .
        \end{align*}
    }
    \addtask{3}{
        We notice that the example knowledge base consists of definite clauses, i.e., clauses with exactly one positive literal.
        We know that the unit propagation algorithm is sound and complete for such knowledge bases, and conclude that the entailed literals are exactly those in $\mathcal{L}$.
        \begin{itemize}
            \item[(i)] As shown in task $2$, $\mathcal{L}$ consists in all positive literals.
            Thus, all positive literals are entailed by the knowledge base, and all negative literals are not entailed by the knowledge base.
            \item[(ii)] Without the clauses $\formulaof{5}$ and $\formulaof{6}$, the knowledge base does not contain any unit clause, and the unit propagation algorithm terminates in $\mathcal{L}=\varnothing$ without a single iteration of the $\mathrm{while}$ loop.
            Thus, no literal is entailed by the knowledge base of the formulas $(0),(1),(2),(3),(4)$.
        \end{itemize}
    }

    \addtask{5}{
    By the result of unit propagation, we know that the contraction of the knowledge base is equivalent to the conjunction of all positive literals.
    There is thus a single model of the knowledge base, which is given by the assignment of all variables to $1$.
    To decide entailment of the test formulas it is enough to check, whether the test formulas are satisfied in this model.
    \begin{itemize}
        \item[i] The test formula $\formulaof{(i)}\equiv\big(\lnot A \lor \lnot B\big)$ is not satisfied in the model, and therefore not entailed by the knowledge base.
        \item[ii] The test formula $\formulaof{(ii)}\equiv\big(\lnot B \lor \lnot M \lor A\big)$ is satisfied in the model (since containing a positive literal), and therefore entailed by the knowledge base.
    \end{itemize}
