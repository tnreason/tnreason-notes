    \exercisehead{3}{Constraint Satisfaction Problems}{May 11, 2026}
    \usecasebox{$n$ queens problem}{
        For arbitrary $n\in\nn$ let there be a grid of $n\times n$ positions with $n$ columns and $n$ rows.
        We want to assign $r$ queens onto positions of the chess board, such that:
        \begin{itemize}
            \item Each column contains exactly one queen.
            \item Each row contains at most one queen.
        \end{itemize}
    }
    \addtask{1}{
        Design the $n$ queens problem as a Constraint Satisfaction Problem.
        \begin{itemize}
            \item[(i)] What are suitable variables, their leg dimensions, and the constraint tensors?
            \item[(ii)] Which hidden variables could be introduced for a sparser representation of the constraints?
        \end{itemize}
        \textit{Hint:} Compare the problem to the Sudoku game.
    }
    \addtask{2}{
        Assume again that we have a partial assignment of $r\leq n$ queens given.
        How many assignments of the remaining $n-r$ queens satisfy the constraints?
        How to compute that number by a tensor network contraction?
    }
    \addtask{3 (Generalized unit propagation)}{
        Assume that we have a partial assignment of $r<n$ queens given. %, such that each pair of queens is assigned to different columns and different rows.
        Would the generalized unit propagation algorithm be able to find a solution for the remaining $r-k$ queens?
    }
    \addtask{4 (Backtracking)}{
        Assume that we have a partial assignment of $r<n$ queens given. %, such that each pair of queens is assigned to different columns and different rows.
        How would the backtracking algorithm with generalized unit propagation as a simplifier find a solution for the remaining $r-k$ queens?
    }
