    \solutionhead{3}{Constraint Satisfaction Problems}{May 11, 2026}
    \addtask{1}{
        \begin{itemize}
            \item[(i)] We enumerate each row by an index $r\in[n]$ and each cell by an index $c\in[n]$.
            We then define for each cell representing tuple $(r,c)$ a variable $\catvariableof{rc}$ with $\catdimof{rc}=2$, indicating whether the cell contains a rook.
            For each row $r\in[n]$ we build a constraint tensor $\hypercoreofat{row,r}{\catvariableof{r\times[n]}}$, where we denote by $\catvariableof{r\times[n]}=\{\catvariableof{rc} \wcols c\in[n]\}$ the variables to a row.
            The constraint tensor is defined for each coordinate $\catindexof{r\times[n]}\in[2]^n$ by
            \begin{align*}
                \hypercoreofat{row,r}{\indexedcatvariableof{r\times[n]}} =
                \begin{cases}
                    1 & \ifspace \sum_{c\in[n]}\catindexof{rc}\leq 1\\
                    0 & \text{else}
                \end{cases} \, .
            \end{align*}
            Vise versa, for each column $c\in[n]$ a constraint tensor
            \begin{align*}
                \hypercoreofat{col,c}{\indexedcatvariableof{[n]\times c}} =
                \begin{cases}
                    1 & \ifspace \sum_{r\in[n]}\catindexof{rc}\leq 1\\
                    0 & \text{else}
                \end{cases} \, .
            \end{align*}
            \item[(ii)] We introduce for each constraint a decomposition variable $\decvariable$ of dimension $n+1$
            \begin{itemize}
                \item For $i\in[n]$ the assignment $\decvariable=i$ shall represent that the $i$th atomic variable has a $1$ assigned.
                \item The state $\decvariable=n$ shall represent that none of the atomic variables has a $1$ assigned.
            \end{itemize}

            For row constraints this decomposition variable is enforced for each $c\in[n]$ by the matrices
            \begin{align*}
                \hypercoreofat{row,r,c}{\catvariableof{rc},\decvariableof{r}^{row}}
                = \tbasisat{\catvariableof{rc}} \otimes \onehotmapofat{c}{\decvariableof{r}^{row}}
                + \fbasisat{\catvariableof{rc}} \otimes \big(\onesat{\decvariableof{r}^{row}}-\onehotmapofat{c}{\decvariableof{r}^{row}}\big) \, .
            \end{align*}
            We then have
            \begin{align*}
                \hypercoreofat{row,r}{\catvariableof{r\times[n]}}
                = \contractionof{
                    \{\hypercoreofat{row,r,c}{\catvariableof{rc},\decvariableof{r}^{row}} \wcols c \in [n]\}
                }{\catvariableof{r\times[n]}} \, .
            \end{align*}
            For column constraints we find analogous decomposition variables, by interchanging the $r$ and $c$ indices.

    \addtask{2}{
        If there is a row or column with two rooks assigned, then there is no assignment satisfying all constraints.
        If not, then there are
        \begin{align*}
            (n-r)! = 1 \cdot 2 \cdot \ldots \cdot (n-r)
        \end{align*}
        possible assignments.
        Lets denote by $E$ the set of the $r$ cell tuples with a rook assigned.
        The total number of assignments respecting all constraints is the contraction
        \begin{align*}
            \contraction{
                \left(\bigcup_{(r,c)\in E} \tbasisat{\catvariableof{rc}} \right)
                \cup
                \left(\bigcup_{r,c\in[n]}
                    \{\hypercoreofat{row,r,c}{\catvariableof{rc},\decvariableof{r}^{row}},\hypercoreofat{col,c,r}{\catvariableof{rc},\decvariableof{c}^{col}}\}
                \right)
            } \, .
        \end{align*}
    }
